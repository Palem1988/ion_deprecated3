93  int dataSize = fileSize - sizeof(uint256);
94  // Don't try to resize to a negative number if file is small
95  if (dataSize < 0)
96  dataSize = 0;

98  vchData.resize(dataSize);
99  uint256 hashIn;
100 
101  // read data and checksum from file
102  try {

105  }
106  catch (std::exception &e) {
107  error("%s : Deserialize or I/O error - %s", __func__, e.what());
108  return HashReadError;

110  filein.fclose();
111

113 
114  // verify stored checksum matches input data
115  uint256 hashTmp = Hash(ssMasternodes.begin(), ssMasternodes.end());
116  if (hashIn != hashTmp)
117  {
118  error("%s : Checksum mismatch, data corrupted", __func__);
119  return IncorrectHash;
120  }
121 
122  unsigned char pchMsgTmp[4];
123  std::string strMagicMessageTmp;

125  // de-serialize file header (masternode cache file specific magic message) and ..

132  error("%s : Invalid masternode cache magic message", __func__);
133  return IncorrectMagicMessage;
134  }
