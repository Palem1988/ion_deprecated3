96 static QString seconds_to_DHMS(quint32 duration)
97 {
98  QString res;
99  int seconds = (int) (duration % 60);
100  duration /= 60;
101  int minutes = (int) (duration % 60);
102  duration /= 60;
103  int hours = (int) (duration % 24);
104  int days = (int) (duration / 24);
105  if((hours == 0)&&(days == 0))
106  return res.sprintf("%02dm:%02ds", minutes, seconds);
107  if (days == 0)
108  return res.sprintf("%02dh:%02dm:%02ds", hours, minutes, seconds);
109  return res.sprintf("%dd %02dh:%02dm:%02ds", days, hours, minutes, seconds);
