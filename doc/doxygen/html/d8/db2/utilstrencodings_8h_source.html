64 template<typename T>
65 std::string HexStr(const T itbegin, const T itend, bool fSpaces=false)
66 {
67  std::string rv;
68  static const char hexmap[16] = { '0', '1', '2', '3', '4', '5', '6', '7',
69  '8', '9', 'a', 'b', 'c', 'd', 'e', 'f' };
70  rv.reserve((itend-itbegin)*3);
71  for(T it = itbegin; it < itend; ++it)
72  {
73  unsigned char val = (unsigned char)(*it);
74  if(fSpaces && it != itbegin)
75  rv.push_back(' ');
76  rv.push_back(hexmap[val>>4]);
77  rv.push_back(hexmap[val&15]);
78  }
79 
80  return rv;
81 }

101 bool TimingResistantEqual(const T& a, const T& b)
102 {
103  if (b.size() == 0) return a.size() == 0;
104  size_t accumulator = a.size() ^ b.size();
105  for (size_t i = 0; i < a.size(); i++)
106  accumulator |= a[i] ^ b[i%b.size()];
107  return accumulator == 0;
108 }
