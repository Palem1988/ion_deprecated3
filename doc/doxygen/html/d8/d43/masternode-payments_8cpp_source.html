28 void ProcessMessageMasternodePayments(CNode* pfrom, std::string& strCommand, CDataStream& vRecv)
29 {
30  if(!stashedSendPool.IsBlockchainSynced()) return;
31 
32  if (strCommand == "mnget") { //Masternode Payments Request Sync
33 
34  if(pfrom->HasFulfilledRequest("mnget")) {
35  LogPrintf("mnget - peer already asked me for the list\n");
36  Misbehaving(pfrom->GetId(), 20);
37  return;
38  }
39 
40  pfrom->FulfilledRequest("mnget");
41  masternodePayments.Sync(pfrom);
42  LogPrintf("mnget - Sent Masternode winners to %s\n", pfrom->addr.ToString().c_str());
43  }
44  else if (strCommand == "mnw") { //Masternode Payments Declare Winner
45 
46  LOCK(cs_masternodepayments);
47 
48  //this is required in litemode
49  CMasternodePaymentWinner winner;
50  vRecv >> winner;
51 
52  if(pindexBest == NULL) return;
53 
54  CTxDestination address1;
55  ExtractDestination(winner.payee, address1);
56  CIonAddress address2(address1);
57 
58  uint256 hash = winner.GetHash();
59  if(mapSeenMasternodeVotes.count(hash)) {
60  if(fDebug) LogPrintf("mnw - seen vote %s Addr %s Height %d bestHeight %d\n", hash.ToString().c_str(), address2.ToString().c_str(), winner.nBlockHeight, pindexBest->nHeight);
61  return;
62  }
63 
64  if(winner.nBlockHeight < pindexBest->nHeight - 10 || winner.nBlockHeight > pindexBest->nHeight+20){
65  LogPrintf("mnw - winner out of range %s Addr %s Height %d bestHeight %d\n", winner.vin.ToString().c_str(), address2.ToString().c_str(), winner.nBlockHeight, pindexBest->nHeight);
66  return;
67  }
68 
69  if(winner.vin.nSequence != std::numeric_limits<unsigned int>::max()){
70  LogPrintf("mnw - invalid nSequence\n");
71  Misbehaving(pfrom->GetId(), 100);
72  return;
73  }
74 
75  LogPrintf("mnw - winning vote - Vin %s Addr %s Height %d bestHeight %d\n", winner.vin.ToString().c_str(), address2.ToString().c_str(), winner.nBlockHeight, pindexBest->nHeight);
76 
77  if(!masternodePayments.CheckSignature(winner)){
78  LogPrintf("mnw - invalid signature\n");
79  Misbehaving(pfrom->GetId(), 100);
80  return;
81  }
82 
83  mapSeenMasternodeVotes.insert(make_pair(hash, winner));
84 
85  if(masternodePayments.AddWinningMasternode(winner)){
86  masternodePayments.Relay(winner);
87  }
88  }
89 }

173 bool CMasternodePayments::AddWinningMasternode(CMasternodePaymentWinner& winnerIn)
174 {
175  uint256 blockHash = 0;
176  if(!GetBlockHash(blockHash, winnerIn.nBlockHeight-576)) {
177  return false;
178  }
179 
180  winnerIn.score = CalculateScore(blockHash, winnerIn.vin);
181 
182  bool foundBlock = false;
183  BOOST_FOREACH(CMasternodePaymentWinner& winner, vWinning){
184  if(winner.nBlockHeight == winnerIn.nBlockHeight) {
185  foundBlock = true;
186  if(winner.score < winnerIn.score){
187  winner.score = winnerIn.score;
188  winner.vin = winnerIn.vin;
189  winner.payee = winnerIn.payee;
190  winner.vchSig = winnerIn.vchSig;
191 
192  mapSeenMasternodeVotes.insert(make_pair(winnerIn.GetHash(), winnerIn));
193 
194  return true;
195  }
196  }
197  }
198 
199  // if it's not in the vector
200  if(!foundBlock){
201  vWinning.push_back(winnerIn);
202  mapSeenMasternodeVotes.insert(make_pair(winnerIn.GetHash(), winnerIn));
203 
204  return true;
205  }
206 
207  return false;

228 bool CMasternodePayments::ProcessBlock(int nBlockHeight)
229 {
230  LOCK(cs_masternodepayments);
231 
232  if(nBlockHeight <= nLastBlockHeight) return false;
233  if(!enabled) return false;
234  CMasternodePaymentWinner newWinner;
235  int nMinimumAge = mnodeman.CountEnabled();
236  CScript payeeSource;
237 
238  uint256 hash;
239  if(!GetBlockHash(hash, nBlockHeight-10)) return false;
240  unsigned int nHash;
241  memcpy(&nHash, &hash, 2);
242 
243  LogPrintf(" ProcessBlock Start nHeight %d - vin %s. \n", nBlockHeight, activeMasternode.vin.ToString().c_str());
244 
245  std::vector<CTxIn> vecLastPayments;
246  BOOST_REVERSE_FOREACH(CMasternodePaymentWinner& winner, vWinning)
247  {
248  //if we already have the same vin - we have one full payment cycle, break
249  if(vecLastPayments.size() > (unsigned int)nMinimumAge) break;
250  vecLastPayments.push_back(winner.vin);
251  }
252 
253  // pay to the oldest MN that still had no payment but its input is old enough and it was active long enough
254  CMasternode *pmn = mnodeman.FindOldestNotInVec(vecLastPayments, nMinimumAge);
255  if(pmn != NULL)
256  {
257  LogPrintf(" Found by FindOldestNotInVec \n");
258 
259  newWinner.score = 0;
260  newWinner.nBlockHeight = nBlockHeight;
261  newWinner.vin = pmn->vin;
262 
263  if(pmn->donationPercentage > 0 && (nHash % 100) <= (unsigned int)pmn->donationPercentage) {
264  newWinner.payee = pmn->donationAddress;
265  } else {
266  newWinner.payee = GetScriptForDestination(pmn->pubkey.GetID());
267  }
268 
269  payeeSource = GetScriptForDestination(pmn->pubkey.GetID());
270  }
271 
272  //if we can't find new MN to get paid, pick first active MN counting back from the end of vecLastPayments list
273  if(newWinner.nBlockHeight == 0 && nMinimumAge > 0)
274  {
275  LogPrintf(" Find by reverse \n");
276 
277  BOOST_REVERSE_FOREACH(CTxIn& vinLP, vecLastPayments)
278  {
279  CMasternode* pmn = mnodeman.Find(vinLP);
280  if(pmn != NULL)
281  {
282  pmn->Check();
283  if(!pmn->IsEnabled()) continue;
284 
285  newWinner.score = 0;
286  newWinner.nBlockHeight = nBlockHeight;
287  newWinner.vin = pmn->vin;
288 
289  if(pmn->donationPercentage > 0 && (nHash % 100) <= (unsigned int)pmn->donationPercentage) {
290  newWinner.payee = pmn->donationAddress;
291  } else {
292  newWinner.payee = GetScriptForDestination(pmn->pubkey.GetID());
293  }
294 
295  payeeSource = GetScriptForDestination(pmn->pubkey.GetID());
296 
297  break; // we found active MN
298  }
299  }
300  }
301 
302  if(newWinner.nBlockHeight == 0) return false;
303 
304  CTxDestination address1;
305  ExtractDestination(newWinner.payee, address1);
306  CIonAddress address2(address1);
307 
308  CTxDestination address3;
309  ExtractDestination(payeeSource, address3);
310  CIonAddress address4(address3);
311 
312  LogPrintf("Winner payee %s nHeight %d vin source %s. \n", address2.ToString().c_str(), newWinner.nBlockHeight, address4.ToString().c_str());
313 
314  if(Sign(newWinner))
315  {
316  if(AddWinningMasternode(newWinner))
317  {
318  Relay(newWinner);
319  nLastBlockHeight = nBlockHeight;
320  return true;
321  }
322  }
323 
324  return false;
325 }
