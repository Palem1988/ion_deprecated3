62 std::string static EncodeDumpString(const std::string &str) {
63  std::stringstream ret;
64  BOOST_FOREACH(unsigned char c, str) {
65  if (c <= 32 || c >= 128 || c == '%') {
66  ret << '%' << HexStr(&c, &c + 1);

68  ret << c;

74 std::string DecodeDumpString(const std::string &str) {
75  std::stringstream ret;
76  for (unsigned int pos = 0; pos < str.length(); pos++) {
77  unsigned char c = str[pos];
78  if (c == '%' && pos+2 < str.length()) {
79  c = (((str[pos+1]>>6)*9+((str[pos+1]-'0')&15)) << 4) |
80  ((str[pos+2]>>6)*9+((str[pos+2]-'0')&15));
81  pos += 2;
82  }
83  ret << c;
84  }
85  return ret.str();
86 }

229 Value importwallet(const Array& params, bool fHelp)
230 {
231  if (fHelp || params.size() != 1)
232  throw runtime_error(
233  "importwallet <filename>\n"
234  "Imports keys from a wallet dump file (see dumpwallet).");
235 
236  EnsureWalletIsUnlocked();
237 
238  ifstream file;
239  file.open(params[0].get_str().c_str());
240  if (!file.is_open())
241  throw JSONRPCError(RPC_INVALID_PARAMETER, "Cannot open wallet dump file");
242 
243  int64_t nTimeBegin = pindexBest->nTime;
244 
245  int64_t nFilesize = std::max((int64_t)1, (int64_t)file.tellg());
246  bool fGood = true;
247 
248  pwalletMain->ShowProgress(_("Importing..."), 0); // show progress dialog in GUI
249  while (file.good()) {
250  pwalletMain->ShowProgress("", std::max(1, std::min(99, (int)(((double)file.tellg() / (double)nFilesize) * 100))));
251  std::string line;
252  std::getline(file, line);
253  if (line.empty() || line[0] == '#')
254  continue;
255 
256  std::vector<std::string> vstr;
257  boost::split(vstr, line, boost::is_any_of(" "));
258  if (vstr.size() < 2)
259  continue;
260  CIonSecret vchSecret;
261  if (!vchSecret.SetString(vstr[0]))
262  continue;
263  CKey key = vchSecret.GetKey();
264  CPubKey pubkey = key.GetPubKey();
265  assert(key.VerifyPubKey(pubkey));
266  CKeyID keyid = pubkey.GetID();
267  if (pwalletMain->HaveKey(keyid)) {
268  LogPrintf("Skipping import of %s (key already present)\n", CIonAddress(keyid).ToString());
269  continue;
270  }
271  int64_t nTime = DecodeDumpTime(vstr[1]);
272  std::string strLabel;
273  bool fLabel = true;
274  for (unsigned int nStr = 2; nStr < vstr.size(); nStr++) {
275  if (boost::algorithm::starts_with(vstr[nStr], "#"))
276  break;
277  if (vstr[nStr] == "change=1")
278  fLabel = false;
279  if (vstr[nStr] == "reserve=1")
280  fLabel = false;
281  if (boost::algorithm::starts_with(vstr[nStr], "label=")) {
282  strLabel = DecodeDumpString(vstr[nStr].substr(6));
283  fLabel = true;
284  }
285  }
286  LogPrintf("Importing %s...\n", CIonAddress(keyid).ToString());
287  if (!pwalletMain->AddKey(key)) {
288  fGood = false;
289  continue;
290  }
291  pwalletMain->mapKeyMetadata[keyid].nCreateTime = nTime;
292  if (fLabel)
293  pwalletMain->SetAddressBookName(keyid, strLabel);
294  nTimeBegin = std::min(nTimeBegin, nTime);
295  }
296  file.close();
297  pwalletMain->ShowProgress("", 100); // hide progress dialog in GUI
298 
299  CBlockIndex *pindex = pindexBest;
300  while (pindex && pindex->pprev && pindex->nTime > nTimeBegin - 7200)
301  pindex = pindex->pprev;
302 
303  if (!pwalletMain->nTimeFirstKey || nTimeBegin < pwalletMain->nTimeFirstKey)
304  pwalletMain->nTimeFirstKey = nTimeBegin;
305 
306  LogPrintf("Rescanning last %i blocks\n", pindexBest->nHeight - pindex->nHeight + 1);
307  pwalletMain->ScanForWalletTransactions(pindex);
308  pwalletMain->ReacceptWalletTransactions();
309  pwalletMain->MarkDirty();
310 
311  if (!fGood)
312  throw JSONRPCError(RPC_WALLET_ERROR, "Error adding some keys to wallet");
313 
314  return Value::null;
315 }

367  vKeyBirth.push_back(std::make_pair(it->second, it->first));

374  file << strprintf("# * Created on %s\n", EncodeDumpTime(GetTime()));

376  file << strprintf("# mined on %s\n", EncodeDumpTime(pindexBest->nTime));
377  file << "\n";
378  for (std::vector<std::pair<int64_t, CKeyID> >::const_iterator it = vKeyBirth.begin(); it != vKeyBirth.end(); it++) {
379  const CKeyID &keyid = it->second;
380  std::string strTime = EncodeDumpTime(it->first);
381  std::string strAddr = CIonAddress(keyid).ToString();
382 
383  CKey key;
384  if (pwalletMain->GetKey(keyid, key)) {
385  if (pwalletMain->mapAddressBook.count(keyid)) {

387  } else if (setKeyPool.count(keyid)) {
388  file << strprintf("%s %s reserve=1 # addr=%s\n", CIonSecret(key).ToString(), strTime, strAddr);
389  } else {
390  file << strprintf("%s %s change=1 # addr=%s\n", CIonSecret(key).ToString(), strTime, strAddr);
391  }
392  }
393  }
394  file << "\n";
395  file << "# End of dump\n";
396  file.close();
397  return Value::null;
