160 public:

168  virtual void Put(const leveldb::Slice& key, const leveldb::Slice& value) {
169  if (key.ToString() == needle) {

171  *deleted = false;

184 // When performing a read, if we have an active batch we need to check it first
185 // before reading from the database, as the rest of the code assumes that once
186 // a database transaction begins reads are consistent with it. It would be good
187 // to change that assumption in future and avoid the performance hit, though in
188 // practice it does not appear to be large.
189 bool CTxDB::ScanBatch(const CDataStream &key, string *value, bool *deleted) const {
190  assert(activeBatch);
191  *deleted = false;
192  CBatchScanner scanner;
193  scanner.needle = key.str();
194  scanner.deleted = deleted;
195  scanner.foundValue = value;
196  leveldb::Status status = activeBatch->Iterate(&scanner);
197  if (!status.ok()) {
198  throw runtime_error(status.ToString());
199  }
200  return scanner.foundEntry;
201 }

311 static CBlockIndex *InsertBlockIndex(uint256 hash)
312 {
313  if (hash == 0)
314  return NULL;
315 
316  // Return existing
317  map<uint256, CBlockIndex*>::iterator mi = mapBlockIndex.find(hash);
318  if (mi != mapBlockIndex.end())
319  return (*mi).second;
320 
321  // Create new
322  CBlockIndex* pindexNew = new CBlockIndex();
323  if (!pindexNew)
324  throw runtime_error("LoadBlockIndex() : new CBlockIndex failed");
325  mi = mapBlockIndex.insert(make_pair(hash, pindexNew)).first;
326  pindexNew->phashBlock = &((*mi).first);
327

331 bool CTxDB::LoadBlockIndex()
332 {
333  if (mapBlockIndex.size() > 0) {
334  // Already loaded once in this session. It can happen during migration
335  // from BDB.
336  return true;
337  }
338  // The block index is an in-memory structure that maps hashes to on-disk
339  // locations where the contents of the block can be found. Here, we scan it
340  // out of the DB and into mapBlockIndex.
341  leveldb::Iterator *iterator = pdb->NewIterator(leveldb::ReadOptions());
342  // Seek to start key.
343  CDataStream ssStartKey(SER_DISK, CLIENT_VERSION);
344  ssStartKey << make_pair(string("blockindex"), uint256(0));
345  iterator->Seek(ssStartKey.str());
346  // Now read each entry.
347  while (iterator->Valid())
348  {
349  boost::this_thread::interruption_point();
350  // Unpack keys and values.
351  CDataStream ssKey(SER_DISK, CLIENT_VERSION);
352  ssKey.write(iterator->key().data(), iterator->key().size());
353  CDataStream ssValue(SER_DISK, CLIENT_VERSION);
354  ssValue.write(iterator->value().data(), iterator->value().size());
355  string strType;
356  ssKey >> strType;
357  // Did we reach the end of the data to read?
358  if (strType != "blockindex")
359  break;
360  CDiskBlockIndex diskindex;
361  ssValue >> diskindex;
362 
363  uint256 blockHash = diskindex.GetBlockHash();
364 
365  // Construct block index object
366  CBlockIndex* pindexNew = InsertBlockIndex(blockHash);
367  pindexNew->pprev = InsertBlockIndex(diskindex.hashPrev);
368  pindexNew->pnext = InsertBlockIndex(diskindex.hashNext);
369  pindexNew->nFile = diskindex.nFile;
370  pindexNew->nBlockPos = diskindex.nBlockPos;
371  pindexNew->nHeight = diskindex.nHeight;
372  pindexNew->nMint = diskindex.nMint;
373  pindexNew->nMoneySupply = diskindex.nMoneySupply;
374  pindexNew->nFlags = diskindex.nFlags;
375  pindexNew->nStakeModifier = diskindex.nStakeModifier;
376  pindexNew->bnStakeModifierV2 = diskindex.bnStakeModifierV2;
377  pindexNew->prevoutStake = diskindex.prevoutStake;
378  pindexNew->nStakeTime = diskindex.nStakeTime;
379  pindexNew->hashProof = diskindex.hashProof;
380  pindexNew->nVersion = diskindex.nVersion;
381  pindexNew->hashMerkleRoot = diskindex.hashMerkleRoot;
382  pindexNew->nTime = diskindex.nTime;
383  pindexNew->nBits = diskindex.nBits;
384  pindexNew->nNonce = diskindex.nNonce;
385 
386  // Watch for genesis block
387  if (pindexGenesisBlock == NULL && blockHash == Params().HashGenesisBlock())
388  pindexGenesisBlock = pindexNew;
389 
390  if (!pindexNew->CheckIndex()) {
391  delete iterator;
392  return error("LoadBlockIndex() : CheckIndex failed at %d", pindexNew->nHeight);
393  }
394 
395  // Ion: build setStakeSeen
396  if (pindexNew->IsProofOfStake())
397  setStakeSeen.insert(make_pair(pindexNew->prevoutStake, pindexNew->nStakeTime));
398 
399  iterator->Next();
400  }
401 
402  delete iterator;
403 
404  boost::this_thread::interruption_point();
405 
406  // Calculate nChainTrust
407  vector<pair<int, CBlockIndex*> > vSortedByHeight;
408  vSortedByHeight.reserve(mapBlockIndex.size());
409  BOOST_FOREACH(const PAIRTYPE(uint256, CBlockIndex*)& item, mapBlockIndex)
410  {
411  CBlockIndex* pindex = item.second;
412  vSortedByHeight.push_back(make_pair(pindex->nHeight, pindex));
413  }
414  sort(vSortedByHeight.begin(), vSortedByHeight.end());
415  BOOST_FOREACH(const PAIRTYPE(int, CBlockIndex*)& item, vSortedByHeight)
416  {
417  CBlockIndex* pindex = item.second;
418  pindex->nChainTrust = (pindex->pprev ? pindex->pprev->nChainTrust : 0) + pindex->GetBlockTrust();
419  }
420 
421  // Load hashBestChain pointer to end of best chain
422  if (!ReadHashBestChain(hashBestChain))
423  {
424  if (pindexGenesisBlock == NULL)
425  return true;
426  return error("CTxDB::LoadBlockIndex() : hashBestChain not loaded");
427  }
428  if (!mapBlockIndex.count(hashBestChain))
429  return error("CTxDB::LoadBlockIndex() : hashBestChain not found in the block index");
430  pindexBest = mapBlockIndex[hashBestChain];
431  nBestHeight = pindexBest->nHeight;
432  nBestChainTrust = pindexBest->nChainTrust;
433 
434  LogPrintf("LoadBlockIndex(): hashBestChain=%s height=%d trust=%s date=%s\n",
435  hashBestChain.ToString(), nBestHeight, nBestChainTrust.ToString(),
436  DateTimeStrFormat("%x %H:%M:%S", pindexBest->GetBlockTime()));
437 
438  // Load bnBestInvalidTrust, OK if it doesn't exist
439  uint256 bnBestInvalidTrust;
440  ReadBestInvalidTrust(bnBestInvalidTrust);
441  nBestInvalidTrust = bnBestInvalidTrust;
442 
443  // Verify blocks in the best chain
444  int nCheckLevel = GetArg("-checklevel", 1);
445  int nCheckDepth = GetArg( "-checkblocks", 500);
446  if (nCheckDepth == 0)
447  nCheckDepth = 1000000000; // suffices until the year 19000
448  if (nCheckDepth > nBestHeight)
449  nCheckDepth = nBestHeight;
450 
451  LogPrintf("Verifying last %i blocks at level %i\n", nCheckDepth, nCheckLevel);
452  CBlockIndex* pindexFork = NULL;
453  map<pair<unsigned int, unsigned int>, CBlockIndex*> mapBlockPos;
454  for (CBlockIndex* pindex = pindexBest; pindex && pindex->pprev; pindex = pindex->pprev)
455  {
456  boost::this_thread::interruption_point();
457  if (pindex->nHeight < nBestHeight-nCheckDepth)
458  break;
459  CBlock block;
460  if (!block.ReadFromDisk(pindex))
461  return error("LoadBlockIndex() : block.ReadFromDisk failed");
462  // check level 1: verify block validity
463  // check level 7: verify block signature too
464  if (nCheckLevel>0 && !block.CheckBlock(true, true, (nCheckLevel>6)))
465  {
466  LogPrintf("LoadBlockIndex() : *** found bad block at %d, hash=%s\n", pindex->nHeight, pindex->GetBlockHash().ToString());
467  pindexFork = pindex->pprev;
468  }
469  // check level 2: verify transaction index validity
470  if (nCheckLevel>1)
471  {
472  pair<unsigned int, unsigned int> pos = make_pair(pindex->nFile, pindex->nBlockPos);
473  mapBlockPos[pos] = pindex;
474  BOOST_FOREACH(const CTransaction &tx, block.vtx)
475  {
476  uint256 hashTx = tx.GetHash();
477  CTxIndex txindex;
478  if (ReadTxIndex(hashTx, txindex))
479  {
480  // check level 3: checker transaction hashes
481  if (nCheckLevel>2 || pindex->nFile != txindex.pos.nFile || pindex->nBlockPos != txindex.pos.nBlockPos)
482  {
483  // either an error or a duplicate transaction
484  CTransaction txFound;
485  if (!txFound.ReadFromDisk(txindex.pos))
486  {
487  LogPrintf("LoadBlockIndex() : *** cannot read mislocated transaction %s\n", hashTx.ToString());
488  pindexFork = pindex->pprev;
489  }
490  else
491  if (txFound.GetHash() != hashTx) // not a duplicate tx
492  {
493  LogPrintf("LoadBlockIndex(): *** invalid tx position for %s\n", hashTx.ToString());
494  pindexFork = pindex->pprev;
495  }
496  }
497  // check level 4: check whether spent txouts were spent within the main chain
498  unsigned int nOutput = 0;
499  if (nCheckLevel>3)
500  {
501  BOOST_FOREACH(const CDiskTxPos &txpos, txindex.vSpent)
502  {
503  if (!txpos.IsNull())
504  {
505  pair<unsigned int, unsigned int> posFind = make_pair(txpos.nFile, txpos.nBlockPos);
506  if (!mapBlockPos.count(posFind))
507  {
508  LogPrintf("LoadBlockIndex(): *** found bad spend at %d, hashBlock=%s, hashTx=%s\n", pindex->nHeight, pindex->GetBlockHash().ToString(), hashTx.ToString());
509  pindexFork = pindex->pprev;
510  }
511  // check level 6: check whether spent txouts were spent by a valid transaction that consume them
512  if (nCheckLevel>5)
513  {
514  CTransaction txSpend;
515  if (!txSpend.ReadFromDisk(txpos))
516  {
517  LogPrintf("LoadBlockIndex(): *** cannot read spending transaction of %s:%i from disk\n", hashTx.ToString(), nOutput);
518  pindexFork = pindex->pprev;
519  }
520  else if (!txSpend.CheckTransaction())
521  {
522  LogPrintf("LoadBlockIndex(): *** spending transaction of %s:%i is invalid\n", hashTx.ToString(), nOutput);
523  pindexFork = pindex->pprev;
524  }
525  else
526  {
527  bool fFound = false;
528  BOOST_FOREACH(const CTxIn &txin, txSpend.vin)
529  if (txin.prevout.hash == hashTx && txin.prevout.n == nOutput)
530  fFound = true;
531  if (!fFound)
532  {
533  LogPrintf("LoadBlockIndex(): *** spending transaction of %s:%i does not spend it\n", hashTx.ToString(), nOutput);
534  pindexFork = pindex->pprev;
535  }
536  }
537  }
538  }
539  nOutput++;
540  }
541  }
542  }
543  // check level 5: check whether all prevouts are marked spent
544  if (nCheckLevel>4)
545  {
546  BOOST_FOREACH(const CTxIn &txin, tx.vin)
547  {
548  CTxIndex txindex;
549  if (ReadTxIndex(txin.prevout.hash, txindex))
550  if (txindex.vSpent.size()-1 < txin.prevout.n || txindex.vSpent[txin.prevout.n].IsNull())
551  {
552  LogPrintf("LoadBlockIndex(): *** found unspent prevout %s:%i in %s\n", txin.prevout.hash.ToString(), txin.prevout.n, hashTx.ToString());
553  pindexFork = pindex->pprev;
554  }
555  }
556  }
557  }
558  }
559  }
560  if (pindexFork)
561  {
562  boost::this_thread::interruption_point();
563  // Reorg back to the fork
564  LogPrintf("LoadBlockIndex() : *** moving best chain pointer back to block %d\n", pindexFork->nHeight);
565  CBlock block;
566  if (!block.ReadFromDisk(pindexFork))
567  return error("LoadBlockIndex() : block.ReadFromDisk failed");
568  CTxDB txdb;
569  block.SetBestChain(txdb, pindexFork);
570  }
571 
572  return true;
573 }
