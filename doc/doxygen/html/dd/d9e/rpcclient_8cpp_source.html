34 Object CallRPC(const string& strMethod, const Array& params)
35 {
36  if (mapArgs["-rpcuser"] == "" && mapArgs["-rpcpassword"] == "")
37  throw runtime_error(strprintf(
38  _("You must set rpcpassword=<password> in the configuration file:\n%s\n"
39  "If the file does not exist, create it with owner-readable-only file permissions."),
40  GetConfigFile().string()));
41 
42  // Connect to localhost
43  bool fUseSSL = GetBoolArg("-rpcssl", false);
44  asio::io_service io_service;
45  ssl::context context(io_service, ssl::context::sslv23);
46  context.set_options(ssl::context::no_sslv2);
47  asio::ssl::stream<asio::ip::tcp::socket> sslStream(io_service, context);
48  SSLIOStreamDevice<asio::ip::tcp> d(sslStream, fUseSSL);
49  iostreams::stream< SSLIOStreamDevice<asio::ip::tcp> > stream(d);
50 
51  bool fWait = GetBoolArg("-rpcwait", false); // -rpcwait means try until server has started
52  do {
53  bool fConnected = d.connect(GetArg("-rpcconnect", "127.0.0.1"), GetArg("-rpcport", itostr(Params().RPCPort())));
54  if (fConnected) break;
55  if (fWait)
56  MilliSleep(1000);
57  else
58  throw runtime_error("couldn't connect to server");
59  } while (fWait);
60 
61  // HTTP basic authentication
62  string strUserPass64 = EncodeBase64(mapArgs["-rpcuser"] + ":" + mapArgs["-rpcpassword"]);
63  map<string, string> mapRequestHeaders;
64  mapRequestHeaders["Authorization"] = string("Basic ") + strUserPass64;
65 
66  // Send request
67  string strRequest = JSONRPCRequest(strMethod, params, 1);
68  string strPost = HTTPPost(strRequest, mapRequestHeaders);
69  stream << strPost << std::flush;
70 
71  // Receive HTTP reply status
72  int nProto = 0;
73  int nStatus = ReadHTTPStatus(stream, nProto);
74 
75  // Receive HTTP reply message headers and body
76  map<string, string> mapHeaders;
77  string strReply;
78  ReadHTTPMessage(stream, mapHeaders, strReply, nProto, MAX_SIZE);
79 
80  if (nStatus == HTTP_UNAUTHORIZED)
81  throw runtime_error("incorrect rpcuser or rpcpassword (authorization failed)");
82  else if (nStatus >= 400 && nStatus != HTTP_BAD_REQUEST && nStatus != HTTP_NOT_FOUND && nStatus != HTTP_INTERNAL_SERVER_ERROR)
83  throw runtime_error(strprintf("server returned HTTP error %d", nStatus));
84  else if (strReply.empty())
85  throw runtime_error("no response from server");
86 
87  // Parse reply
88  Value valReply;
89  if (!read_string(strReply, valReply))
90  throw runtime_error("couldn't parse reply from server");
91  const Object& reply = valReply.get_obj();
92  if (reply.empty())
93  throw runtime_error("expected reply to have result, error and id properties");
94 
95  return reply;
96 }

200 Array RPCConvertValues(const std::string &strMethod, const std::vector<std::string> &strParams)
201 {
202  Array params;
203

217  params.push_back(jVal);

225 int CommandLineRPC(int argc, char *argv[])
226 {
227  string strPrint;
228  int nRet = 0;
229  try
230  {
231  // Skip switches
232  while (argc > 1 && IsSwitchChar(argv[1][0]))
233  {
234  argc--;
235  argv++;
236  }
237 
238  // Method
239  if (argc < 2)
240  throw runtime_error("too few parameters");
241  string strMethod = argv[1];
242 
243  // Parameters default to strings
244  std::vector<std::string> strParams(&argv[2], &argv[argc]);
245  Array params = RPCConvertValues(strMethod, strParams);
246 
247  // Execute
248  Object reply = CallRPC(strMethod, params);
249 
250  // Parse reply
251  const Value& result = find_value(reply, "result");
252  const Value& error = find_value(reply, "error");
253 
254  if (error.type() != null_type)
255  {
256  // Error
257  strPrint = "error: " + write_string(error, false);
258  int code = find_value(error.get_obj(), "code").get_int();
259  nRet = abs(code);
260 
261  if (error.type() != null_type){
262  const Value &errMsg = find_value(error.get_obj(), "message");
263 
264  strPrint = "error code: " + boost::lexical_cast<std::string>(nRet) + "\n";
265 
266  if (!errMsg.is_null())
267  strPrint += "error message: " + errMsg.get_str() + "\n";
268  }
269 
270  }
271  else
272  {
273  // Result
274  if (result.type() == null_type)
275  strPrint = "";
276  else if (result.type() == str_type)
277  strPrint = result.get_str();
278  else
279  strPrint = write_string(result, true);
280  }
281  }
282  catch (boost::thread_interrupted) {
283  throw;
284  }
285  catch (std::exception& e) {
286  strPrint = string("error: ") + e.what();
287  nRet = 87;
288  }
289  catch (...) {
290  PrintException(NULL, "CommandLineRPC()");
291  }
292 
293  if (strPrint != "")
294  {
295  fprintf((nRet == 0 ? stdout : stderr), "%s\n", strPrint.c_str());
296  }
297  return nRet;
298 }
