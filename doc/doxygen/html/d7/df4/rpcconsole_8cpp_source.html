88 bool parseCommandLine(std::vector<std::string> &args, const std::string &strCommand)
89 {
90  enum CmdParseState
91  {
92  STATE_EATING_SPACES,
93  STATE_ARGUMENT,
94  STATE_SINGLEQUOTED,
95  STATE_DOUBLEQUOTED,
96  STATE_ESCAPE_OUTER,
97  STATE_ESCAPE_DOUBLEQUOTED
98  } state = STATE_EATING_SPACES;
99  std::string curarg;
100  foreach(char ch, strCommand)
101  {
102  switch(state)
103  {
104  case STATE_ARGUMENT: // In or after argument
105  case STATE_EATING_SPACES: // Handle runs of whitespace
106  switch(ch)
107  {
108  case '"': state = STATE_DOUBLEQUOTED; break;
109  case '\'': state = STATE_SINGLEQUOTED; break;
110  case '\\': state = STATE_ESCAPE_OUTER; break;
111  case ' ': case '\n': case '\t':
112  if(state == STATE_ARGUMENT) // Space ends argument
113  {
114  args.push_back(curarg);
115  curarg.clear();
116  }
117  state = STATE_EATING_SPACES;
118  break;
119  default: curarg += ch; state = STATE_ARGUMENT;
120  }
121  break;
122  case STATE_SINGLEQUOTED: // Single-quoted string
123  switch(ch)
124  {
125  case '\'': state = STATE_ARGUMENT; break;
126  default: curarg += ch;
127  }
128  break;
129  case STATE_DOUBLEQUOTED: // Double-quoted string
130  switch(ch)
131  {
132  case '"': state = STATE_ARGUMENT; break;
133  case '\\': state = STATE_ESCAPE_DOUBLEQUOTED; break;
134  default: curarg += ch;
135  }
136  break;
137  case STATE_ESCAPE_OUTER: // '\' outside quotes
138  curarg += ch; state = STATE_ARGUMENT;
139  break;
140  case STATE_ESCAPE_DOUBLEQUOTED: // '\' in double-quoted text
141  if(ch != '"' && ch != '\\') curarg += '\\'; // keep '\' for everything but the quote and '\' itself
142  curarg += ch; state = STATE_DOUBLEQUOTED;
143  break;
144  }
145  }
146  switch(state) // final state
147  {
148  case STATE_EATING_SPACES:
149  return true;
150  case STATE_ARGUMENT:
151  args.push_back(curarg);
152  return true;
153  default: // ERROR to end in one of the other states
154  return false;
155  }
156 }

158 void RPCExecutor::request(const QString &command)
159 {
160  std::vector<std::string> args;
161  if(!parseCommandLine(args, command.toStdString()))
162  {
163  emit reply(RPCConsole::CMD_ERROR, QString("Parse error: unbalanced ' or \""));
164  return;
165  }
166  if(args.empty())
167  return; // Nothing to do
168  try
169  {
170  std::string strPrint;
171  // Convert argument list to JSON objects in method-dependent way,
172  // and pass it along with the method name to the dispatcher.
173  json_spirit::Value result = tableRPC.execute(
174  args[0],
175  RPCConvertValues(args[0], std::vector<std::string>(args.begin() + 1, args.end())));
176 
177  // Format result reply
178  if (result.type() == json_spirit::null_type)
179  strPrint = "";
180  else if (result.type() == json_spirit::str_type)
181  strPrint = result.get_str();
182  else
183  strPrint = write_string(result, true);
184 
185  emit reply(RPCConsole::CMD_REPLY, QString::fromStdString(strPrint));
186  }
187  catch (json_spirit::Object& objError)
188  {
189  try // Nice formatting for standard-format error
190  {
191  int code = find_value(objError, "code").get_int();
192  std::string message = find_value(objError, "message").get_str();
193  emit reply(RPCConsole::CMD_ERROR, QString::fromStdString(message) + " (code " + QString::number(code) + ")");
194  }
195  catch(std::runtime_error &) // raised when converting to invalid type, i.e. missing code or message
196  { // Show raw JSON object
197  emit reply(RPCConsole::CMD_ERROR, QString::fromStdString(write_string(json_spirit::Value(objError), false)));
198  }
199  }
200  catch (std::exception& e)
201  {
202  emit reply(RPCConsole::CMD_ERROR, QString("Error: ") + QString::fromStdString(e.what()));
203  }
204 }

245 bool RPCConsole::eventFilter(QObject* obj, QEvent *event)
246 {
247  if(event->type() == QEvent::KeyPress) // Special key handling
248  {
249  QKeyEvent *keyevt = static_cast<QKeyEvent*>(event);
250  int key = keyevt->key();
251  Qt::KeyboardModifiers mod = keyevt->modifiers();
252  switch(key)
253  {
254  case Qt::Key_Up: if(obj == ui->lineEdit) { browseHistory(-1); return true; } break;
255  case Qt::Key_Down: if(obj == ui->lineEdit) { browseHistory(1); return true; } break;
256  case Qt::Key_PageUp: /* pass paging keys to messages widget */
257  case Qt::Key_PageDown:
258  if(obj == ui->lineEdit)
259  {
260  QApplication::postEvent(ui->messagesWidget, new QKeyEvent(*keyevt));
261  return true;
262  }
263  break;
264  default:
265  // Typing in messages widget brings focus to line edit, and redirects key there
266  // Exclude most combinations and keys that emit no text, except paste shortcuts
267  if(obj == ui->messagesWidget && (
268  (!mod && !keyevt->text().isEmpty() && key != Qt::Key_Tab) ||
269  ((mod & Qt::ControlModifier) && key == Qt::Key_V) ||
270  ((mod & Qt::ShiftModifier) && key == Qt::Key_Insert)))
271  {
272  ui->lineEdit->setFocus();
273  QApplication::postEvent(ui->lineEdit, new QKeyEvent(*keyevt));
274  return true;
275  }
276  }
277  }
278  return QWidget::eventFilter(obj, event);
279 }

477 void RPCConsole::on_lineEdit_returnPressed()
478 {
479  QString cmd = ui->lineEdit->text();
480  ui->lineEdit->clear();
481 
482  if(!cmd.isEmpty())
483  {
484  message(CMD_REQUEST, cmd);
485  emit cmdRequest(cmd);
486  // Truncate history from current position
487  history.erase(history.begin() + historyPtr, history.end());
488  // Append command to history
489  history.append(cmd);
490  // Enforce maximum history size
491  while(history.size() > CONSOLE_HISTORY)
492  history.removeFirst();
493  // Set pointer to end of history
494  historyPtr = history.size();
495  // Scroll console view to end
496  scrollToEnd();
497  }
498 }

513 void RPCConsole::startExecutor()
514 {
515  QThread *thread = new QThread;
516  RPCExecutor *executor = new RPCExecutor();
517  executor->moveToThread(thread);
518 
519  // Replies from executor object must go to this object
520  connect(executor, SIGNAL(reply(int,QString)), this, SLOT(message(int,QString)));
521  // Requests from this object must go to executor
522  connect(this, SIGNAL(cmdRequest(QString)), executor, SLOT(request(QString)));
523 
524  // On stopExecutor signal
525  // - queue executor for deletion (in execution thread)
526  // - quit the Qt event loop in the execution thread
527  connect(this, SIGNAL(stopExecutor()), executor, SLOT(deleteLater()));
528  connect(this, SIGNAL(stopExecutor()), thread, SLOT(quit()));
529  // Queue the thread for deletion (in this thread) when it is finished
530  connect(thread, SIGNAL(finished()), thread, SLOT(deleteLater()));
531 
532  // Default implementation of QThread::run() simply spins up an event loop in the thread,
533  // which is what we want.
534  thread->start();
535 }

607 void RPCConsole::peerLayoutChanged()
608 {
609  if (!clientModel || !clientModel->getPeerTableModel())
610  return;
611 
612  const CNodeCombinedStats *stats = NULL;
613  bool fUnselect = false;
614  bool fReselect = false;
615 
616  if (cachedNodeid == -1) // no node selected yet
617  return;
618 
619  // find the currently selected row
620  int selectedRow;
621  QModelIndexList selectedModelIndex = ui->peerWidget->selectionModel()->selectedIndexes();
622  if (selectedModelIndex.isEmpty())
623  selectedRow = -1;
624  else
625  selectedRow = selectedModelIndex.first().row();
626 
627  // check if our detail node has a row in the table (it may not necessarily
628  // be at selectedRow since its position can change after a layout change)
629  int detailNodeRow = clientModel->getPeerTableModel()->getRowByNodeId(cachedNodeid);
630 
631  if (detailNodeRow < 0)
632  {
633  // detail node dissapeared from table (node disconnected)
634  fUnselect = true;
635  cachedNodeid = -1;
636  ui->detailWidget->hide();
637  ui->peerHeading->setText(tr("Select a peer to view detailed information."));
638  }
639  else
640  {
641  if (detailNodeRow != selectedRow)
642  {
643  // detail node moved position
644  fUnselect = true;
645  fReselect = true;
646  }
647 
648  // get fresh stats on the detail node.
