558  void AskFor(const CInv& inv, bool fImmediateRetry = false)
559  {
560  if (mapAskFor.size() > MAPASKFOR_MAX_SZ)
561  return;
562  // a peer may not have multiple non-responded queue positions for a single inv item
563  if (!setAskFor.insert(inv.hash).second)
564  return;
565 
566 
567  // We're using mapAskFor as a priority queue,
568  // the key is the earliest time the request can be sent
569  int64_t nRequestTime;
570  limitedmap<CInv, int64_t>::const_iterator it = mapAlreadyAskedFor.find(inv);
571  if (it != mapAlreadyAskedFor.end())
572  nRequestTime = it->second;
573  else
574  nRequestTime = 0;
575  LogPrint("net", "askfor %s %d (%s)\n", inv.ToString().c_str(), nRequestTime, DateTimeStrFormat("%H:%M:%S", nRequestTime/1000000).c_str());
576 
577  // Make sure not to reuse time indexes to keep things in the same order
578  int64_t nNow = GetTimeMicros() - 1000000;
579  static int64_t nLastTime;
580  ++nLastTime;
581  nNow = std::max(nNow, nLastTime);
582  nLastTime = nNow;
583 
584  // Retry immediately during initial sync otherwise retry 2 minutes after the last
585  if (fImmediateRetry)
586  nRequestTime = nNow;
587  else
588  nRequestTime = std::max(nRequestTime + 2 * 60 * 1000000, nNow);
589  if (it != mapAlreadyAskedFor.end())
590  mapAlreadyAskedFor.update(it, nRequestTime);
591  else
592  mapAlreadyAskedFor.insert(std::make_pair(inv, nRequestTime));
593  mapAskFor.insert(std::make_pair(nRequestTime, inv));
594  }

617  // TODO: Document the precondition of this function. Is cs_vSend locked?
618  void EndMessage() UNLOCK_FUNCTION(cs_vSend)
619  {
620  // The -*messagestest options are intentionally not documented in the help message,
621  // since they are only used during development to debug the networking code and are
622  // not intended for end-users.
623  if (mapArgs.count("-dropmessagestest") && GetRand(GetArg("-dropmessagestest", 2)) == 0)
624  {
625  LogPrint("net", "dropmessages DROPPING SEND MESSAGE\n");
626  AbortMessage();
627  return;
628  }
629 
630  if (ssSend.size() == 0)
631  return;
632 
633  // Set the size
634  unsigned int nSize = ssSend.size() - CMessageHeader::HEADER_SIZE;
635  memcpy((char*)&ssSend[CMessageHeader::MESSAGE_SIZE_OFFSET], &nSize, sizeof(nSize));
636 
637  // Set the checksum
638  uint256 hash = Hash(ssSend.begin() + CMessageHeader::HEADER_SIZE, ssSend.end());
639  unsigned int nChecksum = 0;
640  memcpy(&nChecksum, &hash, sizeof(nChecksum));
641  assert(ssSend.size () >= CMessageHeader::CHECKSUM_OFFSET + sizeof(nChecksum));
642  memcpy((char*)&ssSend[CMessageHeader::CHECKSUM_OFFSET], &nChecksum, sizeof(nChecksum));
643 
644  LogPrint("net", "(%d bytes)\n", nSize);
645 
646  std::deque<CSerializeData>::iterator it = vSendMsg.insert(vSendMsg.end(), CSerializeData());
647  ssSend.GetAndClear(*it);
648  nSendSize += (*it).size();
649 
650  // If write queue empty, attempt "optimistic write"
651  if (it == vSendMsg.begin())
652  SocketSendData(this);
653 
654  LEAVE_CRITICAL_SECTION(cs_vSend);
655  }
