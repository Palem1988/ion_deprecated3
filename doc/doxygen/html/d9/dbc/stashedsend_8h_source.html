111  bool Add(const std::vector<CTxIn> vinIn, int64_t amountIn, const CTransaction collateralIn, const std::vector<CTxOut> voutIn)
112  {
113  if(isSet){return false;}
114 
115  BOOST_FOREACH(const CTxIn& in, vinIn)
116  sev.push_back(in);
117 
118  BOOST_FOREACH(const CTxOut& out, voutIn)
119  vout.push_back(out);
120 
121  amount = amountIn;
122  collateral = collateralIn;
123  isSet = true;
124  addedTime = GetTime();
125 
126  return true;
127  }
128 
129  bool AddSig(const CTxIn& vin)
130  {
131  BOOST_FOREACH(CTxDSIn& s, sev) {
132  if(s.prevout == vin.prevout && s.nSequence == vin.nSequence){
133  if(s.fHasSig){return false;}
134  s.scriptSig = vin.scriptSig;
135  s.prevPubKey = vin.prevPubKey;
136  s.fHasSig = true;
137 
138  return true;
139  }
140  }
141 
142  return false;
143  }
144 
145  bool IsExpired()
146  {
147  return (GetTime() - addedTime) > STASHEDSEND_QUEUE_TIMEOUT;// 120 seconds
148  }
149 };
