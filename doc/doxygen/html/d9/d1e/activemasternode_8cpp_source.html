15 void CActiveMasternode::ManageStatus()
16 {
17  std::string errorMessage;
18 
19  if (fDebug) LogPrintf("CActiveMasternode::ManageStatus() - Begin\n");
20 
21  if(!fMasterNode) return;
22 
23  //need correct adjusted time to send ping
24  bool fIsInitialDownload = IsInitialBlockDownload();
25  if(fIsInitialDownload) {
26  status = MASTERNODE_SYNC_IN_PROCESS;
27  LogPrintf("CActiveMasternode::ManageStatus() - Sync in progress. Must wait until sync is complete to start masternode.\n");
28  return;
29  }
30 
31  if(status == MASTERNODE_INPUT_TOO_NEW || status == MASTERNODE_NOT_CAPABLE || status == MASTERNODE_SYNC_IN_PROCESS){
32  status = MASTERNODE_NOT_PROCESSED;
33  }
34 
35  if(status == MASTERNODE_NOT_PROCESSED) {
36  if(strMasterNodeAddr.empty()) {
37  if(!GetLocal(service)) {
38  notCapableReason = "Can't detect external address. Please use the masternodeaddr configuration option.";
39  status = MASTERNODE_NOT_CAPABLE;
40  LogPrintf("CActiveMasternode::ManageStatus() %s (%s)\n", getStatusMessage(status), notCapableReason.c_str());
41  return;
42  }
43  } else {
44  service = CService(strMasterNodeAddr, true);
45  }
46 
47  LogPrintf("CActiveMasternode::ManageStatus() - Checking inbound connection to '%s'\n", service.ToString().c_str());
48 
49 
50  if(!ConnectNode((CAddress)service, service.ToString().c_str())){
51  notCapableReason = "Could not connect to " + service.ToString();
52  status = MASTERNODE_NOT_CAPABLE;
53  LogPrintf("CActiveMasternode::ManageStatus() - not capable: %s\n", notCapableReason.c_str());
54  return;
55  }
56 
57 
58  if(pwalletMain->IsLocked()){
59  notCapableReason = "Wallet is locked.";
60  status = MASTERNODE_NOT_CAPABLE;
61  LogPrintf("CActiveMasternode::ManageStatus() - not capable: %s\n", notCapableReason.c_str());
62  return;
63  }
64 
65  // Set defaults
66  status = MASTERNODE_NOT_CAPABLE;
67  notCapableReason = "Unknown. Check debug.log for more information.\n";
68 
69  // Choose coins to use
70  CPubKey pubKeyCollateralAddress;
71  CKey keyCollateralAddress;
72 
73  if(GetMasterNodeVin(vin, pubKeyCollateralAddress, keyCollateralAddress)) {
74 
75  if(GetInputAge(vin) < MASTERNODE_MIN_CONFIRMATIONS){
76  notCapableReason = "Input must have least " + boost::lexical_cast<string>(MASTERNODE_MIN_CONFIRMATIONS) +
77  " confirmations - " + boost::lexical_cast<string>(GetInputAge(vin)) + " confirmations";
78  LogPrintf("CActiveMasternode::ManageStatus() - %s\n", notCapableReason.c_str());
79  status = MASTERNODE_INPUT_TOO_NEW;
80  return;
81  }
82 
83  LogPrintf("CActiveMasternode::ManageStatus() - Is capable master node!\n");
84 
85  status = MASTERNODE_IS_CAPABLE;
86  notCapableReason = "";
87 
88  pwalletMain->LockCoin(vin.prevout);
89 
90  // send to all nodes
91  CPubKey pubKeyMasternode;
92  CKey keyMasternode;
93 
94  if(!stashedSendSigner.SetKey(strMasterNodePrivKey, errorMessage, keyMasternode, pubKeyMasternode))
95  {
96  LogPrintf("ActiveMasternode::Dseep() - Error upon calling SetKey: %s\n", errorMessage.c_str());
97  return;
98  }
99 
100  /* donations are not supported in ioncoin.conf */
101  CScript donationAddress = CScript();
102  int donationPercentage = 0;
103 
104  if(!Register(vin, service, keyCollateralAddress, pubKeyCollateralAddress, keyMasternode, pubKeyMasternode, donationAddress, donationPercentage, errorMessage)) {
105  LogPrintf("CActiveMasternode::ManageStatus() - Error on Register: %s\n", errorMessage.c_str());
106  }
107 
108  return;
109  } else {
110  notCapableReason = "Could not find suitable coins!";
111  LogPrintf("CActiveMasternode::ManageStatus() - Could not find suitable coins!\n");
112  }
113  }
114 
115  //send to all peers
116  if(!Dseep(errorMessage)) {
117  LogPrintf("CActiveMasternode::ManageStatus() - Error on Ping: %s\n", errorMessage.c_str());
118  }
119 }

182 bool CActiveMasternode::Dseep(CTxIn vin, CService service, CKey keyMasternode, CPubKey pubKeyMasternode, std::string &retErrorMessage, bool stop) {
183  std::string errorMessage;
184  std::vector<unsigned char> vchMasterNodeSignature;
185  std::string strMasterNodeSignMessage;
186  int64_t masterNodeSignatureTime = GetAdjustedTime();
187 
188  std::string strMessage = service.ToString() + boost::lexical_cast<std::string>(masterNodeSignatureTime) + boost::lexical_cast<std::string>(stop);
189 
190  if(!stashedSendSigner.SignMessage(strMessage, errorMessage, vchMasterNodeSignature, keyMasternode)) {
191  retErrorMessage = "sign message failed: " + errorMessage;
192  LogPrintf("CActiveMasternode::Dseep() - Error: %s\n", retErrorMessage.c_str());
193  return false;
194  }
195 
196  if(!stashedSendSigner.VerifyMessage(pubKeyMasternode, vchMasterNodeSignature, strMessage, errorMessage)) {
197  retErrorMessage = "Verify message failed: " + errorMessage;
198  LogPrintf("CActiveMasternode::Dseep() - Error: %s\n", retErrorMessage.c_str());
199  return false;
200  }
201 
202  // Update Last Seen timestamp in masternode list
203  CMasternode* pmn = mnodeman.Find(vin);
204  if(pmn != NULL)
205  {
206  if(stop)
207  mnodeman.Remove(pmn->vin);
208  else
209  pmn->UpdateLastSeen();
210  } else {
211  // Seems like we are trying to send a ping while the masternode is not registered in the network
212  retErrorMessage = "Stashedsend Masternode List doesn't include our masternode, Shutting down masternode pinging service! " + vin.ToString();
213  LogPrintf("CActiveMasternode::Dseep() - Error: %s\n", retErrorMessage.c_str());
214  status = MASTERNODE_NOT_CAPABLE;
215  notCapableReason = retErrorMessage;
216  return false;
217  }
218 
219  //send to all peers
220  LogPrintf("CActiveMasternode::Dseep() - RelayMasternodeEntryPing vin = %s\n", vin.ToString().c_str());
221  mnodeman.RelayMasternodeEntryPing(vin, vchMasterNodeSignature, masterNodeSignatureTime, stop);
222 
223  return true;
224 }

273 bool CActiveMasternode::Register(CTxIn vin, CService service, CKey keyCollateralAddress, CPubKey pubKeyCollateralAddress, CKey keyMasternode, CPubKey pubKeyMasternode, CScript donationAddress, int donationPercentage, std::string &retErrorMessage) {
274  std::string errorMessage;

281 
282  std::string strMessage = service.ToString() + boost::lexical_cast<std::string>(masterNodeSignatureTime) + vchPubKey + vchPubKey2 + boost::lexical_cast<std::string>(PROTOCOL_VERSION) + donationAddress.ToString() + boost::lexical_cast<std::string>(donationPercentage);
283

285  retErrorMessage = "sign message failed: " + errorMessage;

287  return false;
288  }
289 
290  if(!stashedSendSigner.VerifyMessage(pubKeyCollateralAddress, vchMasterNodeSignature, strMessage, errorMessage)) {
291  retErrorMessage = "Verify message failed: " + errorMessage;
292  LogPrintf("CActiveMasternode::Register() - Error: %s\n", retErrorMessage.c_str());
293  return false;
294  }
295 
296  CMasternode* pmn = mnodeman.Find(vin);
297  if(pmn == NULL)
298  {

301  mn.UpdateLastSeen(masterNodeSignatureTime);
302  mnodeman.Add(mn);
303  }
304 
305  //send to all peers
306  LogPrintf("CActiveMasternode::Register() - RelayElectionEntry vin = %s\n", vin.ToString().c_str());
307  mnodeman.RelayMasternodeEntry(vin, service, vchMasterNodeSignature, masterNodeSignatureTime, pubKeyCollateralAddress, pubKeyMasternode, -1, -1, masterNodeSignatureTime, PROTOCOL_VERSION, donationAddress, donationPercentage);
308 
309  return true;
310 }

316 bool CActiveMasternode::GetMasterNodeVin(CTxIn& vin, CPubKey& pubkey, CKey& secretKey, std::string strTxHash, std::string strOutputIndex) {
317  CScript pubScript;
318 
319  // Find possible candidates
320  vector<COutput> possibleCoins = SelectCoinsMasternode();
321  COutput *selectedOutput;
322 
323  // Find the vin
324  if(!strTxHash.empty()) {
325  // Let's find it
326  uint256 txHash(strTxHash);
327  int outputIndex = boost::lexical_cast<int>(strOutputIndex);
328  bool found = false;
329  BOOST_FOREACH(COutput& out, possibleCoins) {
330  if(out.tx->GetHash() == txHash && out.i == outputIndex)
331  {
332  selectedOutput = &out;
333  found = true;
334  break;
335  }
336  }
337  if(!found) {
338  LogPrintf("CActiveMasternode::GetMasterNodeVin - Could not locate valid vin\n");
339  return false;
340  }
341  } else {
342  // No output specified, Select the first one
343  if(possibleCoins.size() > 0) {
344  selectedOutput = &possibleCoins[0];
345  } else {
346  LogPrintf("CActiveMasternode::GetMasterNodeVin - Could not locate specified vin from possible list\n");
347  return false;
348  }
349  }
350 
351  // At this point we have a selected output, retrieve the associated info
352  return GetVinFromOutput(*selectedOutput, vin, pubkey, secretKey);
353 }
