67  template <typename Stream, typename Operation>
68  inline void SerializationOp(Stream& s, Operation ser_action, int nType, int nVersion) {
69  std::string sAddress = address.toStdString();
70  std::string sLabel = label.toStdString();
71  std::string sMessage = message.toStdString();
72 
73  unsigned int nSerSize = 0;
74  READWRITE(this->nVersion);
75  nVersion = this->nVersion;
76  READWRITE(sAddress);
77  READWRITE(sLabel);
78  READWRITE(amount);
79  READWRITE(sMessage);
80 
81  if (ser_action.ForRead())
82  {
83  address = QString::fromStdString(sAddress);
84  label = QString::fromStdString(sLabel);
85  message = QString::fromStdString(sMessage);
86  }
