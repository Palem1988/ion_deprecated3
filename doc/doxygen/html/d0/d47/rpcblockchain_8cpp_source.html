17 double GetDifficulty(const CBlockIndex* blockindex)
18 {
19  // Floating point number that is a multiple of the minimum difficulty,
20  // minimum difficulty = 1.0.
21  if (blockindex == NULL)
22  {
23  if (pindexBest == NULL)
24  return 1.0;
25  else
26  blockindex = GetLastBlockIndex(pindexBest, false);
27  }
28 
29  int nShift = (blockindex->nBits >> 24) & 0xff;
30 
31  double dDiff =
32  (double)0x0000ffff / (double)(blockindex->nBits & 0x00ffffff);
33 
34  while (nShift < 29)
35  {
36  dDiff *= 256.0;
37  nShift++;
38  }
39  while (nShift > 29)
40  {
41  dDiff /= 256.0;
42  nShift--;
43  }
44 
45  return dDiff;
46 }

48 double GetPoWMHashPS()
49 {
50  if (pindexBest->nHeight >= Params().LastPOWBlock())
51  return 0;
52 
53  int nPoWInterval = 72;
54  int64_t nTargetSpacingWorkMin = 30, nTargetSpacingWork = 30;
55 
56  CBlockIndex* pindex = pindexGenesisBlock;
57  CBlockIndex* pindexPrevWork = pindexGenesisBlock;
58 
59  while (pindex)
60  {
61  if (pindex->IsProofOfWork())
62  {
63  int64_t nActualSpacingWork = pindex->GetBlockTime() - pindexPrevWork->GetBlockTime();
64  nTargetSpacingWork = ((nPoWInterval - 1) * nTargetSpacingWork + nActualSpacingWork + nActualSpacingWork) / (nPoWInterval + 1);
65  nTargetSpacingWork = max(nTargetSpacingWork, nTargetSpacingWorkMin);
66  pindexPrevWork = pindex;
67  }
68 
69  pindex = pindex->pnext;
70  }
71 
72  return GetDifficulty() * 4294.967296 / nTargetSpacingWork;
73 }

75 double GetPoSKernelPS()
76 {
77  int nPoSInterval = 72;
78  double dStakeKernelsTriedAvg = 0;
79  int nStakesHandled = 0, nStakesTime = 0;
80 
81  CBlockIndex* pindex = pindexBest;;
82  CBlockIndex* pindexPrevStake = NULL;
83 
84  while (pindex && nStakesHandled < nPoSInterval)
85  {
86  if (pindex->IsProofOfStake())
87  {
88  if (pindexPrevStake)
89  {
90  dStakeKernelsTriedAvg += GetDifficulty(pindexPrevStake) * 4294967296.0;
91  nStakesTime += pindexPrevStake->nTime - pindex->nTime;
92  nStakesHandled++;
93  }
94  pindexPrevStake = pindex;
95  }
96 
97  pindex = pindex->pprev;
98  }
99 
100  double result = 0;
101 
102  if (nStakesTime)
103  result = dStakeKernelsTriedAvg / nStakesTime;
104 
105  result *= STAKE_TIMESTAMP_MASK + 1;
106 
107  return result;
108 }
