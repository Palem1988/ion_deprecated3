52  if(!IsBlockchainSynced()) return;
53 
54  if (strCommand == "dsa") { //StashedSend Accept Into Pool
55  if (pfrom->nVersion < MIN_POOL_PEER_PROTO_VERSION) {
56  std::string strError = _("Incompatible version.");
57  LogPrintf("dsa -- incompatible version! \n");
58  pfrom->PushMessage("dssu", sessionID, GetState(), GetEntriesCount(), MASTERNODE_REJECTED, strError);
59 
60  return;
61  }
62 
63  if(!fMasterNode){
64  std::string strError = _("This is not a Masternode.");
65  LogPrintf("dsa -- not a Masternode! \n");
66  pfrom->PushMessage("dssu", sessionID, GetState(), GetEntriesCount(), MASTERNODE_REJECTED, strError);
67 
68  return;
69  }
70 
71  int nDenom;
72  CTransaction txCollateral;
73  vRecv >> nDenom >> txCollateral;
74 
75  std::string error = "";
76  CMasternode* pmn = mnodeman.Find(activeMasternode.vin);
77  if(pmn == NULL)
78  {
79  std::string strError = _("Not in the Masternode list.");
80  pfrom->PushMessage("dssu", sessionID, GetState(), GetEntriesCount(), MASTERNODE_REJECTED, strError);
81  return;
82  }
83 
84  if(sessionUsers == 0) {
85  if(pmn->nLastDsq != 0 &&
86  pmn->nLastDsq + mnodeman.CountMasternodesAboveProtocol(MIN_POOL_PEER_PROTO_VERSION)/5 > mnodeman.nDsqCount){
87  LogPrintf("dsa -- last dsq too recent, must wait. %s \n", pfrom->addr.ToString().c_str());
88  std::string strError = _("Last Stashedsend was too recent.");
89  pfrom->PushMessage("dssu", sessionID, GetState(), GetEntriesCount(), MASTERNODE_REJECTED, strError);
90  return;
91  }
92  }
93 
94  if(!IsCompatibleWithSession(nDenom, txCollateral, error))
95  {
96  LogPrintf("dsa -- not compatible with existing transactions! \n");
97  pfrom->PushMessage("dssu", sessionID, GetState(), GetEntriesCount(), MASTERNODE_REJECTED, error);
98  return;
99  } else {
100  LogPrintf("dsa -- is compatible, please submit! \n");
101  pfrom->PushMessage("dssu", sessionID, GetState(), GetEntriesCount(), MASTERNODE_ACCEPTED, error);
102  return;
103  }
104  } else if (strCommand == "dsq") { //Stashedsend Queue
105  TRY_LOCK(cs_stashedsend, lockRecv);
106  if(!lockRecv) return;
107 
108  if (pfrom->nVersion < MIN_POOL_PEER_PROTO_VERSION) {
109  return;
110  }
111 
112  CStashedsendQueue dsq;
113  vRecv >> dsq;
114 
115 
116  CService addr;
117  if(!dsq.GetAddress(addr)) return;
118  if(!dsq.CheckSignature()) return;
119 
120  if(dsq.IsExpired()) return;
121 
122  CMasternode* pmn = mnodeman.Find(dsq.vin);
123  if(pmn == NULL) return;
124 
125  // if the queue is ready, submit if we can
126  if(dsq.ready) {
127  if(!pSubmittedToMasternode) return;
128  if((CNetAddr)pSubmittedToMasternode->addr != (CNetAddr)addr){
129  LogPrintf("dsq - message doesn't match current Masternode - %s != %s\n", pSubmittedToMasternode->addr.ToString().c_str(), addr.ToString().c_str());
130  return;
131  }
132 
133  if(state == POOL_STATUS_QUEUE){
134  LogPrintf("Stashedsend queue is ready - %s\n", addr.ToString().c_str());
135  PrepareStashedsendDenominate();
136  }
137  } else {
138  BOOST_FOREACH(CStashedsendQueue q, vecStashedsendQueue){
139  if(q.vin == dsq.vin) return;
140  }
141 
142  LogPrint("stashedsend", "dsq last %d last2 %d count %d\n", pmn->nLastDsq, pmn->nLastDsq + mnodeman.size()/5, mnodeman.nDsqCount);
143  //don't allow a few nodes to dominate the queuing process
144  if(pmn->nLastDsq != 0 &&
145  pmn->nLastDsq + mnodeman.CountMasternodesAboveProtocol(MIN_POOL_PEER_PROTO_VERSION)/5 > mnodeman.nDsqCount){
146  LogPrint("stashedsend", "dsq -- Masternode sending too many dsq messages. %s \n", pmn->addr.ToString().c_str());
147  return;
148  }
149  mnodeman.nDsqCount++;
150  pmn->nLastDsq = mnodeman.nDsqCount;
151  pmn->allowFreeTx = true;
152 
153  LogPrint("stashedsend", "dsq - new Stashedsend queue object - %s\n", addr.ToString().c_str());
154  vecStashedsendQueue.push_back(dsq);
155  dsq.Relay();
156  dsq.time = GetTime();
157  }
158 
159  } else if (strCommand == "dsi") { //StashedSend vIn
160  std::string error = "";
161  if (pfrom->nVersion < MIN_POOL_PEER_PROTO_VERSION) {
162  LogPrintf("dsi -- incompatible version! \n");
163  error = _("Incompatible version.");
164  pfrom->PushMessage("dssu", sessionID, GetState(), GetEntriesCount(), MASTERNODE_REJECTED, error);
165 
166  return;
167  }
168 
169  if(!fMasterNode){
170  LogPrintf("dsi -- not a Masternode! \n");
171  error = _("This is not a Masternode.");
172  pfrom->PushMessage("dssu", sessionID, GetState(), GetEntriesCount(), MASTERNODE_REJECTED, error);
173 
174  return;
175  }
176 
177  std::vector<CTxIn> in;
178  CAmount nAmount;
179  CTransaction txCollateral;
180  std::vector<CTxOut> out;
181  vRecv >> in >> nAmount >> txCollateral >> out;
182 
183  //do we have enough users in the current session?
184  if(!IsSessionReady()){
185  LogPrintf("dsi -- session not complete! \n");
186  error = _("Session not complete!");
187  pfrom->PushMessage("dssu", sessionID, GetState(), GetEntriesCount(), MASTERNODE_REJECTED, error);
188  return;
189  }
190 
191  //do we have the same denominations as the current session?
192  if(!IsCompatibleWithEntries(out))
193  {
194  LogPrintf("dsi -- not compatible with existing transactions! \n");
195  error = _("Not compatible with existing transactions.");
196  pfrom->PushMessage("dssu", sessionID, GetState(), GetEntriesCount(), MASTERNODE_REJECTED, error);
197  return;
198  }
199 
200  //check it like a transaction
201  {
202  CAmount nValueIn = 0;
203  CAmount nValueOut = 0;
204  bool missingTx = false;
205 
206  CValidationState state;
207  CTransaction tx;
208 
209  BOOST_FOREACH(const CTxOut o, out){
210  nValueOut += o.nValue;
211  tx.vout.push_back(o);
212 
213  if(o.scriptPubKey.size() != 25){
214  LogPrintf("dsi - non-standard pubkey detected! %s\n", o.scriptPubKey.ToString().c_str());
215  error = _("Non-standard public key detected.");
216  pfrom->PushMessage("dssu", sessionID, GetState(), GetEntriesCount(), MASTERNODE_REJECTED, error);
217  return;
218  }
219  if(!o.scriptPubKey.IsNormalPaymentScript()){
220  LogPrintf("dsi - invalid script! %s\n", o.scriptPubKey.ToString().c_str());
221  error = _("Invalid script detected.");
222  pfrom->PushMessage("dssu", sessionID, GetState(), GetEntriesCount(), MASTERNODE_REJECTED, error);
223  return;
224  }
225  }
226 
227  BOOST_FOREACH(const CTxIn i, in){
228  tx.vin.push_back(i);
229 
230  LogPrint("stashedsend", "dsi -- tx in %s\n", i.ToString().c_str());
231 
232  CTransaction tx2;
233  uint256 hash;
234  if(GetTransaction(i.prevout.hash, tx2, hash)){
235  if(tx2.vout.size() > i.prevout.n) {
236  nValueIn += tx2.vout[i.prevout.n].nValue;
237  }
238  } else{
239  missingTx = true;
240  }
241  }
242 
243  if (nValueIn > STASHEDSEND_POOL_MAX) {
244  LogPrintf("dsi -- more than Stashedsend pool max! %s\n", tx.ToString().c_str());
245  error = _("Value more than Stashedsend pool maximum allows.");
246  pfrom->PushMessage("dssu", sessionID, GetState(), GetEntriesCount(), MASTERNODE_REJECTED, error);
247  return;
248  }
249 
250  if(!missingTx){
251  if (nValueIn-nValueOut > nValueIn*.01) {
252  LogPrintf("dsi -- fees are too high! %s\n", tx.ToString().c_str());
253  error = _("Transaction fees are too high.");
254  pfrom->PushMessage("dssu", sessionID, GetState(), GetEntriesCount(), MASTERNODE_REJECTED, error);
255  return;
256  }
257  } else {
258  LogPrintf("dsi -- missing input tx! %s\n", tx.ToString().c_str());
259  error = _("Missing input transaction information.");
260  pfrom->PushMessage("dssu", sessionID, GetState(), GetEntriesCount(), MASTERNODE_REJECTED, error);
261  return;
262  }
263 
264  if(!AcceptableInputs(mempool, tx, false, NULL, false, true)){
265  LogPrintf("dsi -- transaction not valid! \n");
266  error = _("Transaction not valid.");
267  pfrom->PushMessage("dssu", sessionID, GetState(), GetEntriesCount(), MASTERNODE_REJECTED, error);
268  return;
269  }
270  }
271 
272  if(AddEntry(in, nAmount, txCollateral, out, error)){
273  pfrom->PushMessage("dssu", sessionID, GetState(), GetEntriesCount(), MASTERNODE_ACCEPTED, error);
274  Check();
275 
276  RelayStatus(sessionID, GetState(), GetEntriesCount(), MASTERNODE_RESET);
277  } else {
278  pfrom->PushMessage("dssu", sessionID, GetState(), GetEntriesCount(), MASTERNODE_REJECTED, error);
279  }
280  } else if (strCommand == "dssu") { //Stashedsend status update
281  if (pfrom->nVersion < MIN_POOL_PEER_PROTO_VERSION) {
282  return;
283  }
284  if(!pSubmittedToMasternode) return;
285  if((CNetAddr)pSubmittedToMasternode->addr != (CNetAddr)pfrom->addr){
286  //LogPrintf("dssu - message doesn't match current Masternode - %s != %s\n", pSubmittedToMasternode->addr.ToString().c_str(), pfrom->addr.ToString().c_str());
287  return;
288  }
289 
290  int sessionIDMessage;
291  int state;
292  int entriesCount;
293  int accepted;
294  std::string error;
295  vRecv >> sessionIDMessage >> state >> entriesCount >> accepted >> error;
296 
297  LogPrint("stashedsend", "dssu - state: %i entriesCount: %i accepted: %i error: %s \n", state, entriesCount, accepted, error.c_str());
298 
299  if((accepted != 1 && accepted != 0) && sessionID != sessionIDMessage){
300  LogPrintf("dssu - message doesn't match current Stashedsend session %d %d\n", sessionID, sessionIDMessage);
301  return;
302  }
303 
304  StatusUpdate(state, entriesCount, accepted, error, sessionIDMessage);
305 
306  } else if (strCommand == "dss") { //StashedSend Sign Final Tx
307  if (pfrom->nVersion < MIN_POOL_PEER_PROTO_VERSION) {
308  return;
309  }
310 
311  vector<CTxIn> sigs;
312  vRecv >> sigs;
313 
314  bool success = false;
315  int count = 0;
316 
317  BOOST_FOREACH(const CTxIn item, sigs)
318  {
319  if(AddScriptSig(item)) success = true;
320  LogPrint("stashedsend", " -- sigs count %d %d\n", (int)sigs.size(), count);
321  count++;
322  }
323 
324  if(success){
325  stashedSendPool.Check();
326  RelayStatus(stashedSendPool.sessionID, stashedSendPool.GetState(), stashedSendPool.GetEntriesCount(), MASTERNODE_RESET);
327  }
328  } else if (strCommand == "dsf") { //Stashedsend Final tx
329  if (pfrom->nVersion < MIN_POOL_PEER_PROTO_VERSION) {
330  return;
331  }
332  if(!pSubmittedToMasternode) return;
333  if((CNetAddr)pSubmittedToMasternode->addr != (CNetAddr)pfrom->addr){
334  //LogPrintf("dsc - message doesn't match current Masternode - %s != %s\n", pSubmittedToMasternode->addr.ToString().c_str(), pfrom->addr.ToString().c_str());
335  return;
336  }
337  int sessionIDMessage;
338  CTransaction txNew;
339  vRecv >> sessionIDMessage >> txNew;
340  if(sessionID != sessionIDMessage){
341  LogPrint("stashedsend", "dsf - message doesn't match current stashedsend session %d %d\n", sessionID, sessionIDMessage);
342  return;
343  }
344  //check to see if input is spent already? (and probably not confirmed)
345  SignFinalTransaction(txNew, pfrom);
346  } else if (strCommand == "dsc") { //Stashedsend Complete
347  if (pfrom->nVersion < MIN_POOL_PEER_PROTO_VERSION) {
348  return;
349  }
350 
351  if(!pSubmittedToMasternode) return;
352  if((CNetAddr)pSubmittedToMasternode->addr != (CNetAddr)pfrom->addr){
353  //LogPrintf("dsc - message doesn't match current Masternode - %s != %s\n", pSubmittedToMasternode->addr.ToString().c_str(), pfrom->addr.ToString().c_str());
354  return;
355  }
356 
357  int sessionIDMessage;
358  bool error;
359  int errorID;
360  vRecv >> sessionIDMessage >> error >> errorID;
361 
362  if(sessionID != sessionIDMessage){
363  LogPrint("stashedsend", "dsc - message doesn't match current stashedsend session %d %d\n", stashedSendPool.sessionID, sessionIDMessage);
364  return;
365  }
366 
367  stashedSendPool.CompletedTransaction(error, errorID);
368  }

468 std::string CStashedsendPool::GetStatus()
469 {
470  static int showingStashedSendMessage = 0;
471  showingStashedSendMessage+=10;
472  std::string suffix = "";
473 
474  if(pindexBest->nHeight - cachedLastSuccess < minBlockSpacing ||!IsBlockchainSynced()) {
475  return strAutoDenomResult;
476  }
477  switch(state) {
478  case POOL_STATUS_IDLE:
479  return _("Stashedsend is idle.");
480  case POOL_STATUS_ACCEPTING_ENTRIES:
481  if(entriesCount == 0) {
482  showingStashedSendMessage = 0;
483  return strAutoDenomResult;
484  } else if (lastEntryAccepted == 1) {
485  if(showingStashedSendMessage % 10 > 8) {
486  lastEntryAccepted = 0;
487  showingStashedSendMessage = 0;
488  }
489  return _("Stashedsend request complete:") + " " + _("Your transaction was accepted into the pool!");
490  } else {
491  std::string suffix = "";
492  if( showingStashedSendMessage % 70 <= 40) return strprintf(_("Submitted following entries to masternode: %u / %d"), entriesCount, GetMaxPoolTransactions());
493  else if(showingStashedSendMessage % 70 <= 50) suffix = ".";
494  else if(showingStashedSendMessage % 70 <= 60) suffix = "..";
495  else if(showingStashedSendMessage % 70 <= 70) suffix = "...";
496  return strprintf(_("Submitted to masternode, waiting for more entries ( %u / %d ) %s"), entriesCount, GetMaxPoolTransactions(), suffix);
497  }
498  case POOL_STATUS_SIGNING:
499  if( showingStashedSendMessage % 70 <= 40) return _("Found enough users, signing ...");
500  else if(showingStashedSendMessage % 70 <= 50) suffix = ".";
501  else if(showingStashedSendMessage % 70 <= 60) suffix = "..";
502  else if(showingStashedSendMessage % 70 <= 70) suffix = "...";
503  return strprintf(_("Found enough users, signing ( waiting %s )"), suffix);
504  case POOL_STATUS_TRANSMISSION:
505  return _("Transmitting final transaction.");
506  case POOL_STATUS_FINALIZE_TRANSACTION:
507  return _("Finalizing transaction.");
508  case POOL_STATUS_ERROR:
509  return _("Stashedsend request incomplete:") + " " + lastMessage + " " + _("Will retry...");
510  case POOL_STATUS_SUCCESS:
511  return _("Stashedsend request complete:") + " " + lastMessage;
512  case POOL_STATUS_QUEUE:
513  if( showingStashedSendMessage % 70 <= 30) suffix = ".";
514  else if(showingStashedSendMessage % 70 <= 50) suffix = "..";
515  else if(showingStashedSendMessage % 70 <= 70) suffix = "...";
516  return strprintf(_("Submitted to masternode, waiting in queue %s"), suffix);;
517  default:
518  return strprintf(_("Unknown state: id = %u"), state);
519  }
520 }

525 void CStashedsendPool::Check()
526 {
527  if(fMasterNode) LogPrint("stashedsend", "CStashedsendPool::Check() - entries count %lu\n", entries.size());
528  //printf("CStashedsendPool::Check() %d - %d - %d\n", state, anonTx.CountEntries(), GetTimeMillis()-lastTimeChanged);
529 
530  if(fMasterNode) {
531  LogPrint("stashedsend", "CStashedsendPool::Check() - entries count %lu\n", entries.size());
532  // If entries is full, then move on to the next phase
533  if(state == POOL_STATUS_ACCEPTING_ENTRIES && (int)entries.size() >= GetMaxPoolTransactions())
534  {
535  LogPrint("stashedsend", "CStashedsendPool::Check() -- TRYING TRANSACTION \n");
536  UpdateState(POOL_STATUS_FINALIZE_TRANSACTION);
537  }
538  }
539 
540  // create the finalized transaction for distribution to the clients
541  if(state == POOL_STATUS_FINALIZE_TRANSACTION) {
542  LogPrint("stashedsend", "CStashedsendPool::Check() -- FINALIZE TRANSACTIONS\n");
543  UpdateState(POOL_STATUS_SIGNING);
544 
545  if (fMasterNode) {
546  CTransaction txNew;
547 
548  // make our new transaction
549  for(unsigned int i = 0; i < entries.size(); i++){
550  BOOST_FOREACH(const CTxOut& v, entries[i].vout)
551  txNew.vout.push_back(v);
552 
553  BOOST_FOREACH(const CTxDSIn& s, entries[i].sev)
554  txNew.vin.push_back(s);
555  }
556 
557  // shuffle the outputs for improved anonymity
558  std::random_shuffle ( txNew.vin.begin(), txNew.vin.end(), randomizeList);
559  std::random_shuffle ( txNew.vout.begin(), txNew.vout.end(), randomizeList);
560 
561 
562  LogPrint("stashedsend", "Transaction 1: %s\n", txNew.ToString());
563  finalTransaction = txNew;
564 
565  // request signatures from clients
566  RelayFinalTransaction(sessionID, finalTransaction);
567  }
568  }
569 
570  // If we have all of the signatures, try to compile the transaction
571  if(fMasterNode && state == POOL_STATUS_SIGNING && SignaturesComplete()) {
572  LogPrint("stashedsend", "CStashedsendPool::Check() -- SIGNING\n");
573  UpdateState(POOL_STATUS_TRANSMISSION);
574  CheckFinalTransaction();
575  }
576 
577  // reset if we're here for 10 seconds
578  if((state == POOL_STATUS_ERROR || state == POOL_STATUS_SUCCESS) && GetTimeMillis()-lastTimeChanged >= 10000) {
579  LogPrint("stashedsend", "CStashedsendPool::Check() -- timeout, RESETTING\n");
580  UnlockCoins();
581  SetNull();
582  if(fMasterNode) RelayStatus(sessionID, GetState(), GetEntriesCount(), MASTERNODE_RESET);
583  }
584 }
