56 // select a block from the candidate blocks in vSortedByTimestamp, excluding
57 // already selected blocks in vSelectedBlocks, and with timestamp up to
58 // nSelectionIntervalStop.
59 static bool SelectBlockFromCandidates(vector<pair<int64_t, uint256> >& vSortedByTimestamp, map<uint256, const CBlockIndex*>& mapSelectedBlocks,
60  int64_t nSelectionIntervalStop, uint64_t nStakeModifierPrev, const CBlockIndex** pindexSelected)
61 {
62  bool fSelected = false;
63  uint256 hashBest = 0;
64  *pindexSelected = (const CBlockIndex*) 0;
65  BOOST_FOREACH(const PAIRTYPE(int64_t, uint256)& item, vSortedByTimestamp)
66  {
67  if (!mapBlockIndex.count(item.second))
68  return error("SelectBlockFromCandidates: failed to find block index for candidate block %s", item.second.ToString());
69  const CBlockIndex* pindex = mapBlockIndex[item.second];
70  if (fSelected && pindex->GetBlockTime() > nSelectionIntervalStop)
71  break;
72  if (mapSelectedBlocks.count(pindex->GetBlockHash()) > 0)
73  continue;
74  // compute the selection hash by hashing its proof-hash and the
75  // previous proof-of-stake modifier
76  CDataStream ss(SER_GETHASH, 0);
77  ss << pindex->hashProof << nStakeModifierPrev;
78  uint256 hashSelection = Hash(ss.begin(), ss.end());
79  // the selection hash is divided by 2**32 so that proof-of-stake block
80  // is always favored over proof-of-work block. this is to preserve
81  // the energy efficiency property
82  if (pindex->IsProofOfStake())
83  hashSelection >>= 32;
84  if (fSelected && hashSelection < hashBest)
85  {
86  hashBest = hashSelection;
87  *pindexSelected = (const CBlockIndex*) pindex;
88  }
89  else if (!fSelected)
90  {
91  fSelected = true;
92  hashBest = hashSelection;
93  *pindexSelected = (const CBlockIndex*) pindex;
94  }
95  }
96  LogPrint("stakemodifier", "SelectBlockFromCandidates: selection hash=%s\n", hashBest.ToString());
97  return fSelected;
98 }

100 // Stake Modifier (hash modifier of proof-of-stake):
101 // The purpose of stake modifier is to prevent a txout (coin) owner from
102 // computing future proof-of-stake generated by this txout at the time
103 // of transaction confirmation. To meet kernel protocol, the txout
104 // must hash with a future stake modifier to generate the proof.
105 // Stake modifier consists of bits each of which is contributed from a
106 // selected block of a given block group in the past.
107 // The selection of a block is based on a hash of the block's proof-hash and
108 // the previous stake modifier.
109 // Stake modifier is recomputed at a fixed time interval instead of every
110 // block. This is to make it difficult for an attacker to gain control of
111 // additional bits in the stake modifier, even after generating a chain of
112 // blocks.
113 bool ComputeNextStakeModifier(const CBlockIndex* pindexPrev, uint64_t& nStakeModifier, bool& fGeneratedStakeModifier)
114 {
115  nStakeModifier = 0;
116  fGeneratedStakeModifier = false;
117  if (!pindexPrev)
118  {
119  fGeneratedStakeModifier = true;
120  return true; // genesis block's modifier is 0
121  }
122  // First find current stake modifier and its generation block time
123  // if it's not old enough, return the same stake modifier
124  int64_t nModifierTime = 0;
125  if (!GetLastStakeModifier(pindexPrev, nStakeModifier, nModifierTime))
126  return error("ComputeNextStakeModifier: unable to get last modifier");
127  LogPrint("stakemodifier", "ComputeNextStakeModifier: prev modifier=0x%016x time=%s\n", nStakeModifier, DateTimeStrFormat(nModifierTime));
128  if (nModifierTime / nModifierInterval >= pindexPrev->GetBlockTime() / nModifierInterval)
129  return true;
130 
131  // Sort candidate blocks by timestamp
132  vector<pair<int64_t, uint256> > vSortedByTimestamp;
133  vSortedByTimestamp.reserve(64 * nModifierInterval / Params().TargetSpacing());
134 
135  int64_t nSelectionInterval = GetStakeModifierSelectionInterval();
136  int64_t nSelectionIntervalStart = (pindexPrev->GetBlockTime() / nModifierInterval) * nModifierInterval - nSelectionInterval;
137  const CBlockIndex* pindex = pindexPrev;
138  while (pindex && pindex->GetBlockTime() >= nSelectionIntervalStart)
139  {
140  vSortedByTimestamp.push_back(make_pair(pindex->GetBlockTime(), pindex->GetBlockHash()));
141  pindex = pindex->pprev;
142  }
143  int nHeightFirstCandidate = pindex ? (pindex->nHeight + 1) : 0;
144  reverse(vSortedByTimestamp.begin(), vSortedByTimestamp.end());
145  sort(vSortedByTimestamp.begin(), vSortedByTimestamp.end());
146 
147  // Select 64 blocks from candidate blocks to generate stake modifier
148  uint64_t nStakeModifierNew = 0;
149  int64_t nSelectionIntervalStop = nSelectionIntervalStart;
150  map<uint256, const CBlockIndex*> mapSelectedBlocks;
151  for (int nRound=0; nRound<min(64, (int)vSortedByTimestamp.size()); nRound++)
152  {
153  // add an interval section to the current selection round
154  nSelectionIntervalStop += GetStakeModifierSelectionIntervalSection(nRound);
155  // select a block from the candidates of current round
156  if (!SelectBlockFromCandidates(vSortedByTimestamp, mapSelectedBlocks, nSelectionIntervalStop, nStakeModifier, &pindex))
157  return error("ComputeNextStakeModifier: unable to select block at round %d", nRound);
158  // write the entropy bit of the selected block
159  nStakeModifierNew |= (((uint64_t)pindex->GetStakeEntropyBit()) << nRound);
160  // add the selected block from candidates to selected list
161  mapSelectedBlocks.insert(make_pair(pindex->GetBlockHash(), pindex));
162  LogPrint("stakemodifier", "ComputeNextStakeModifier: selected round %d stop=%s height=%d bit=%d\n", nRound, DateTimeStrFormat(nSelectionIntervalStop), pindex->nHeight, pindex->GetStakeEntropyBit());
163  }
164 
165  // Print selection map for visualization of the selected blocks
166  if (LogAcceptCategory("stakemodifier"))
167  {
168  string strSelectionMap = "";
169  // '-' indicates proof-of-work blocks not selected
170  strSelectionMap.insert(0, pindexPrev->nHeight - nHeightFirstCandidate + 1, '-');
171  pindex = pindexPrev;
172  while (pindex && pindex->nHeight >= nHeightFirstCandidate)
173  {
174  // '=' indicates proof-of-stake blocks not selected
175  if (pindex->IsProofOfStake())
176  strSelectionMap.replace(pindex->nHeight - nHeightFirstCandidate, 1, "=");
177  pindex = pindex->pprev;
178  }
179  BOOST_FOREACH(const PAIRTYPE(uint256, const CBlockIndex*)& item, mapSelectedBlocks)
180  {
181  // 'S' indicates selected proof-of-stake blocks
182  // 'W' indicates selected proof-of-work blocks
183  strSelectionMap.replace(item.second->nHeight - nHeightFirstCandidate, 1, item.second->IsProofOfStake()? "S" : "W");
184  }
185  LogPrintf("ComputeNextStakeModifier: selection height [%d, %d] map %s\n", nHeightFirstCandidate, pindexPrev->nHeight, strSelectionMap);
186  }
187  LogPrint("stakemodifier", "ComputeNextStakeModifier: new modifier=0x%016x time=%s\n", nStakeModifierNew, DateTimeStrFormat(pindexPrev->GetBlockTime()));
188 
189  nStakeModifier = nStakeModifierNew;
190  fGeneratedStakeModifier = true;
191  return true;
192 }

209 bool CheckStakeKernelHash(CBlockIndex* pindexPrev, unsigned int nBits, unsigned int nTimeBlockFrom, const CTransaction& txPrev, const COutPoint& prevout, unsigned int nTimeTx, uint256& hashProofOfStake, uint256& targetProofOfStake, bool fPrintProofOfStake)
210 {
211  if (nTimeTx < txPrev.nTime) // Transaction timestamp violation
212  return error("CheckStakeKernelHash() : nTime violation");
213 
214  // Base target
215  uint256 bnTarget;
216  bnTarget.SetCompact(nBits);
217 
218  // Weighted target
219  CAmount nValueIn = txPrev.vout[prevout.n].nValue;
220  uint256 bnWeight = nValueIn;
221  bnTarget *= bnWeight;
222 
223  targetProofOfStake = bnTarget;
224 
225  uint64_t nStakeModifier = pindexPrev->nStakeModifier;
226  uint256 bnStakeModifierV2 = pindexPrev->bnStakeModifierV2;
227  int nStakeModifierHeight = pindexPrev->nHeight;
228  int64_t nStakeModifierTime = pindexPrev->nTime;
229 
230  // Calculate hash
231  CDataStream ss(SER_GETHASH, 0);
232 
233  ss << bnStakeModifierV2;
234  ss << txPrev.nTime << prevout.hash << prevout.n << nTimeTx;
235  hashProofOfStake = Hash(ss.begin(), ss.end());
236 
237  if (fPrintProofOfStake)
238  {
239  LogPrintf("CheckStakeKernelHash() : using modifier 0x%016x at height=%d timestamp=%s for block from timestamp=%s\n",
240  nStakeModifier, nStakeModifierHeight,
241  DateTimeStrFormat(nStakeModifierTime),
242  DateTimeStrFormat(nTimeBlockFrom));
243  LogPrintf("CheckStakeKernelHash() : check modifier=0x%016x nTimeBlockFrom=%u nTimeTxPrev=%u nPrevout=%u nTimeTx=%u hashProof=%s\n",
244  nStakeModifier,
245  nTimeBlockFrom, txPrev.nTime, prevout.n, nTimeTx,
246  hashProofOfStake.ToString());
247  }
248 
249  // Now check if proof-of-stake hash meets target protocol
250  if (hashProofOfStake > bnTarget){
251  return false;
252  }
253 
254  if (fDebug && !fPrintProofOfStake)
255  {
256  LogPrintf("CheckStakeKernelHash() : using modifier 0x%016x at height=%d timestamp=%s for block from timestamp=%s\n",
257  nStakeModifier, nStakeModifierHeight,
258  DateTimeStrFormat(nStakeModifierTime),
259  DateTimeStrFormat(nTimeBlockFrom));
260  LogPrintf("CheckStakeKernelHash() : pass modifier=0x%016x nTimeBlockFrom=%u nTimeTxPrev=%u nPrevout=%u nTimeTx=%u hashProof=%s\n",
261  nStakeModifier,
262  nTimeBlockFrom, txPrev.nTime, prevout.n, nTimeTx,
263  hashProofOfStake.ToString());
264  }
265 
266  return true;
267 }

270 bool CheckProofOfStake(CBlockIndex* pindexPrev, const CTransaction& tx, unsigned int nBits, uint256& hashProofOfStake, uint256& targetProofOfStake)
271 {
272  if (!tx.IsCoinStake())
273  return error("CheckProofOfStake() : called on non-coinstake %s", tx.GetHash().ToString());
274 
275  // Kernel (input 0) must match the stake hash target per coin age (nBits)
276  const CTxIn& txin = tx.vin[0];
277 
278  // First try finding the previous transaction in database
279  CTxDB txdb("r");
280  CTransaction txPrev;
281  CTxIndex txindex;
282  if (!txPrev.ReadFromDisk(txdb, txin.prevout, txindex))
283  return tx.DoS(1, error("CheckProofOfStake() : INFO: read txPrev failed")); // previous transaction not in main chain, may occur during initial download
284 
285  // Verify signature
286  if (!VerifySignature(txPrev, tx, 0, SCRIPT_VERIFY_NONE, 0))
287  return tx.DoS(100, error("CheckProofOfStake() : VerifySignature failed on coinstake %s", tx.GetHash().ToString()));
288 
289  // Read block header
290  CBlock block;
291  if (!block.ReadFromDisk(txindex.pos.nFile, txindex.pos.nBlockPos, false))
292  return fDebug? error("CheckProofOfStake() : read block failed") : false; // unable to read block of previous transaction
293 
294  int nDepth;
295  if (IsConfirmedInNPrevBlocks(txindex, pindexPrev, DetermineCoinbaseMaturity() - 1, nDepth))
296  return tx.DoS(100, error("CheckProofOfStake() : tried to stake at depth %d", nDepth + 1));
297 
298  if (!CheckStakeKernelHash(pindexPrev, nBits, block.GetBlockTime(), txPrev, txin.prevout, tx.nTime, hashProofOfStake, targetProofOfStake, fDebug))
299  return tx.DoS(1, error("CheckProofOfStake() : INFO: check kernel failed on coinstake %s, hashProof=%s", tx.GetHash().ToString(), hashProofOfStake.ToString())); // may occur during initial download or if behind on block chain sync
300 
301  return true;
302 }
