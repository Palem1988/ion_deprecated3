30 void ProcessSpork(CNode* pfrom, std::string& strCommand, CDataStream& vRecv)
31 {
32  if(fLiteMode) return; //disable all stashedsend/masternode related functionality
33 
34  if (strCommand == "spork")
35  {
36  //LogPrintf("ProcessSpork::spork\n");
37  CDataStream vMsg(vRecv);
38  CSporkMessage spork;
39  vRecv >> spork;
40 
41  if(pindexBest == NULL) return;
42 
43  uint256 hash = spork.GetHash();
44  if(mapSporksActive.count(spork.nSporkID)) {
45  if(mapSporksActive[spork.nSporkID].nTimeSigned >= spork.nTimeSigned){
46  if(fDebug) LogPrintf("spork - seen %s block %d \n", hash.ToString().c_str(), pindexBest->nHeight);
47  return;
48  } else {
49  if(fDebug) LogPrintf("spork - got updated spork %s block %d \n", hash.ToString().c_str(), pindexBest->nHeight);
50  }
51  }
52 
53  LogPrintf("spork - new %s ID %d Time %d bestHeight %d\n", hash.ToString().c_str(), spork.nSporkID, spork.nValue, pindexBest->nHeight);
54 
55  if(!sporkManager.CheckSignature(spork)){
56  LogPrintf("spork - invalid signature\n");
57  Misbehaving(pfrom->GetId(), 100);
58  return;
59  }
60 
61  mapSporks[hash] = spork;
62  mapSporksActive[spork.nSporkID] = spork;
63  sporkManager.Relay(spork);
64 
65  //does a task if needed
66  ExecuteSpork(spork.nSporkID, spork.nValue);
67  }
68  if (strCommand == "getsporks")
69  {
70  std::map<int, CSporkMessage>::iterator it = mapSporksActive.begin();
71 
72  while(it != mapSporksActive.end()) {
73  pfrom->PushMessage("spork", it->second);
74  it++;
75  }
76  }
77
