164 
165  list<string> laddedNodes(0);
166  if (params.size() == 1)
167  {
168  LOCK(cs_vAddedNodes);
169  BOOST_FOREACH(string& strAddNode, vAddedNodes)
170  laddedNodes.push_back(strAddNode);
171  }
172  else
173  {
174  string strNode = params[1].get_str();
175  LOCK(cs_vAddedNodes);
176  BOOST_FOREACH(string& strAddNode, vAddedNodes)
177  if (strAddNode == strNode)
178  {
179  laddedNodes.push_back(strAddNode);
180  break;
181  }
182  if (laddedNodes.size() == 0)
183  throw JSONRPCError(RPC_CLIENT_NODE_NOT_ADDED, "Error: Node has not been added.");
184  }
185 
186  if (!fDns)
187  {
188  Object ret;
189  BOOST_FOREACH(string& strAddNode, laddedNodes)
190  ret.push_back(Pair("addednode", strAddNode));
191  return ret;
192  }
193 
194  Array ret;
195 
196  list<pair<string, vector<CService> > > laddedAddreses(0);
197  BOOST_FOREACH(string& strAddNode, laddedNodes)
198  {
199  vector<CService> vservNode(0);
200  if(Lookup(strAddNode.c_str(), vservNode, Params().GetDefaultPort(), fNameLookup, 0))
201  laddedAddreses.push_back(make_pair(strAddNode, vservNode));
202  else
203  {
204  Object obj;
205  obj.push_back(Pair("addednode", strAddNode));
206  obj.push_back(Pair("connected", false));
207  Array addresses;
208  obj.push_back(Pair("addresses", addresses));
209  }
210  }
211 
212  LOCK(cs_vNodes);
213  for (list<pair<string, vector<CService> > >::iterator it = laddedAddreses.begin(); it != laddedAddreses.end(); it++)
214  {
215  Object obj;
216  obj.push_back(Pair("addednode", it->first));
217 
218  Array addresses;
219  bool fConnected = false;
220  BOOST_FOREACH(CService& addrNode, it->second)
221  {
222  bool fFound = false;
223  Object node;
224  node.push_back(Pair("address", addrNode.ToString()));
225  BOOST_FOREACH(CNode* pnode, vNodes)
226  if (pnode->addr == addrNode)
227  {
228  fFound = true;
229  fConnected = true;
230  node.push_back(Pair("connected", pnode->fInbound ? "inbound" : "outbound"));
231  break;
232  }
233  if (!fFound)
234  node.push_back(Pair("connected", "false"));
235  addresses.push_back(node);
236  }
237  obj.push_back(Pair("connected", fConnected));
238  obj.push_back(Pair("addresses", addresses));
239  ret.push_back(obj);
240  }
241 
242  return ret;
243 }

249 Value sendalert(const Array& params, bool fHelp)
250 {
251  if (fHelp || params.size() < 6)
252  throw runtime_error(
253  "sendalert <message> <privatekey> <minver> <maxver> <priority> <id> [cancelupto]\n"
254  "<message> is the alert text message\n"
255  "<privatekey> is hex string of alert master private key\n"
256  "<minver> is the minimum applicable internal client version\n"
257  "<maxver> is the maximum applicable internal client version\n"
258  "<priority> is integer priority number\n"
259  "<id> is the alert id\n"
260  "[cancelupto] cancels all alert id's up to this number\n"
261  "Returns true or false.");
262 
263  CAlert alert;
264  CKey key;
265 
266  alert.strStatusBar = params[0].get_str();
267  alert.nMinVer = params[2].get_int();
268  alert.nMaxVer = params[3].get_int();
269  alert.nPriority = params[4].get_int();
270  alert.nID = params[5].get_int();
271  if (params.size() > 6)
272  alert.nCancel = params[6].get_int();
273  alert.nVersion = PROTOCOL_VERSION;
274  alert.nRelayUntil = GetAdjustedTime() + 365*24*60*60;
275  alert.nExpiration = GetAdjustedTime() + 365*24*60*60;
276 
277  CDataStream sMsg(SER_NETWORK, PROTOCOL_VERSION);
278  sMsg << (CUnsignedAlert)alert;
279  alert.vchMsg = vector<unsigned char>(sMsg.begin(), sMsg.end());
280 
281  vector<unsigned char> vchPrivKey = ParseHex(params[1].get_str());
282  key.SetPrivKey(CPrivKey(vchPrivKey.begin(), vchPrivKey.end()), false); // if key is not correct openssl may crash
283  if (!key.Sign(Hash(alert.vchMsg.begin(), alert.vchMsg.end()), alert.vchSig))
284  throw runtime_error(
285  "Unable to sign alert, check private key?\n");
286  if(!alert.ProcessAlert())
287  throw runtime_error(
288  "Failed to process alert.\n");
289  // Relay alert
290  {
291  LOCK(cs_vNodes);
292  BOOST_FOREACH(CNode* pnode, vNodes)
293  alert.RelayTo(pnode);
294  }
295 
296  Object result;
297  result.push_back(Pair("strStatusBar", alert.strStatusBar));
298  result.push_back(Pair("nVersion", alert.nVersion));
299  result.push_back(Pair("nMinVer", alert.nMinVer));
300  result.push_back(Pair("nMaxVer", alert.nMaxVer));
301  result.push_back(Pair("nPriority", alert.nPriority));
302  result.push_back(Pair("nID", alert.nID));
303  if (alert.nCancel > 0)
304  result.push_back(Pair("nCancel", alert.nCancel));
305  return result;
306 }

323 Value setban(const Array& params, bool fHelp)
324 {
325  string strCommand;
326  if (params.size() >= 2)
327  strCommand = params[1].get_str();
328  if (fHelp || params.size() < 2 ||
329  (strCommand != "add" && strCommand != "remove"))
330  throw runtime_error(
331  "setban \"ip(/netmask)\" \"add|remove\" (bantime) (absolute)\n"
332  "\nAttempts add or remove a IP/Subnet from the banned list.\n"
333  "\nArguments:\n"
334  "1. \"ip(/netmask)\" (string, required) The IP/Subnet (see getpeerinfo for nodes ip) with a optional netmask (default is /32 = single ip)\n"
335  "2. \"command\" (string, required) 'add' to add a IP/Subnet to the list, 'remove' to remove a IP/Subnet from the list\n"
336  "3. \"bantime\" (numeric, optional) time in seconds how long (or until when if [absolute] is set) the ip is banned (0 or empty means using the default time of 24h which can also be overwritten by the -bantime startup argument)\n"
337  "4. \"absolute\" (boolean, optional) If set, the bantime must be a absolute timestamp in seconds since epoch (Jan 1 1970 GMT)\n"
338  "\nExamples:\n"
339  + HelpExampleCli("setban", "\"192.168.0.6\" \"add\" 86400")
340  + HelpExampleCli("setban", "\"192.168.0.0/24\" \"add\"")
341  + HelpExampleRpc("setban", "\"192.168.0.6\", \"add\" 86400")
342  );
343 
344  CSubNet subNet;
345  CNetAddr netAddr;
346  bool isSubnet = false;
347 
348  if (params[0].get_str().find("/") != string::npos)
349  isSubnet = true;
350 
351  if (!isSubnet)
352  netAddr = CNetAddr(params[0].get_str());
353  else
354  subNet = CSubNet(params[0].get_str());
355 
356  if (! (isSubnet ? subNet.IsValid() : netAddr.IsValid()) )
357  throw JSONRPCError(RPC_CLIENT_NODE_ALREADY_ADDED, "Error: Invalid IP/Subnet");
358 
359  if (strCommand == "add")
360  {
361  if (isSubnet ? CNode::IsBanned(subNet) : CNode::IsBanned(netAddr))
362  throw JSONRPCError(RPC_CLIENT_NODE_ALREADY_ADDED, "Error: IP/Subnet already banned");
363 
364  int64_t banTime = 0; //use standard bantime if not specified
365  if (params.size() >= 3 && !params[2].is_null())
366  banTime = params[2].get_int64();
367 
368  bool absolute = false;
369  if (params.size() == 4)
370  absolute = params[3].get_bool();
371 
372  isSubnet ? CNode::Ban(subNet, BanReasonManuallyAdded, banTime, absolute) : CNode::Ban(netAddr, BanReasonManuallyAdded, banTime, absolute);
373 
374  //disconnect possible nodes
375  while(CNode *bannedNode = (isSubnet ? FindNode(subNet) : FindNode(netAddr)))
376  bannedNode->CloseSocketDisconnect();
377  }
378  else if(strCommand == "remove")
379  {
380  if (!( isSubnet ? CNode::Unban(subNet) : CNode::Unban(netAddr) ))
381  throw JSONRPCError(RPC_MISC_ERROR, "Error: Unban failed");
382  }
383 
384  DumpBanlist(); //store banlist to disk
385  uiInterface.BannedListChanged();
386 
387  return Value::null;
388 }
