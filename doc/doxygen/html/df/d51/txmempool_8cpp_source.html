28 bool CTxMemPool::addUnchecked(const uint256& hash, CTransaction &tx)
29 {
30  // Add to memory pool without checking anything.
31  // Used by main.cpp AcceptToMemoryPool(), which DOES do
32  // all the appropriate checks.
33  LOCK(cs);
34  {
35  mapTx[hash] = tx;
36  for (unsigned int i = 0; i < tx.vin.size(); i++)
37  mapNextTx[tx.vin[i].prevout] = CInPoint(&mapTx[hash], i);
38  nTransactionsUpdated++;
39  }
40  return true;
41 }

43 bool CTxMemPool::remove(const CTransaction &tx, bool fRecursive)
44 {
45  // Remove transaction from memory pool
46  {
47  LOCK(cs);
48  uint256 hash = tx.GetHash();
49  if (mapTx.count(hash))
50  {
51  if (fRecursive) {
52  for (unsigned int i = 0; i < tx.vout.size(); i++) {
53  std::map<COutPoint, CInPoint>::iterator it = mapNextTx.find(COutPoint(hash, i));
54  if (it != mapNextTx.end())
55  remove(*it->second.ptx, true);
56  }
57  }
58  BOOST_FOREACH(const CTxIn& txin, tx.vin)
59  mapNextTx.erase(txin.prevout);
60  mapTx.erase(hash);
61  nTransactionsUpdated++;
62  }
63  }
64  return true;
65 }

67 bool CTxMemPool::removeConflicts(const CTransaction &tx)
68 {
69  // Remove transactions which depend on inputs of tx, recursively
70  LOCK(cs);
71  BOOST_FOREACH(const CTxIn &txin, tx.vin) {
72  std::map<COutPoint, CInPoint>::iterator it = mapNextTx.find(txin.prevout);
73  if (it != mapNextTx.end()) {
74  const CTransaction &txConflict = *it->second.ptx;
75  if (txConflict != tx)
76  remove(txConflict, true);
77  }
78  }
79  return true;
80 }
