91 {
92  int nRequired = params[0].get_int();
93  const Array& keys = params[1].get_array();
94 
95  // Gather public keys
96  if (nRequired < 1)
97  throw runtime_error("a multisignature address must require at least one key to redeem");
98  if ((int)keys.size() < nRequired)
99  throw runtime_error(
100  strprintf("not enough keys supplied "
101  "(got %" PRIszu" keys, but need at least %d to redeem)", keys.size(), nRequired));
102  std::vector<CPubKey> pubkeys;
103  pubkeys.resize(keys.size());
104  for (unsigned int i = 0; i < keys.size(); i++)
105  {
106  const std::string& ks = keys[i].get_str();
107 #ifdef ENABLE_WALLET
108  // Case 1: Ion address and we have full public key:
109  CIonAddress address(ks);
110  if (pwalletMain && address.IsValid())
111  {
112  CKeyID keyID;
113  if (!address.GetKeyID(keyID))
114  throw runtime_error(
115  strprintf("%s does not refer to a key",ks));
116  CPubKey vchPubKey;
117  if (!pwalletMain->GetPubKey(keyID, vchPubKey))
118  throw runtime_error(
119  strprintf("no full public key for address %s",ks));
120  if (!vchPubKey.IsFullyValid())
121  throw runtime_error(" Invalid public key: "+ks);
122  pubkeys[i] = vchPubKey;
123  }
124 
125  // Case 2: hex public key
126  else
127 #endif
128  if (IsHex(ks))
129  {
130  CPubKey vchPubKey(ParseHex(ks));
131  if (!vchPubKey.IsFullyValid())
132  throw runtime_error(" Invalid public key: "+ks);
133  pubkeys[i] = vchPubKey;
134  }
135  else
136  {
137  throw runtime_error(" Invalid public key: "+ks);
138  }
139  }
140  CScript result;
141  result.SetMultisig(nRequired, pubkeys);
142  return result;
143 }

255 {

257 
258  CAccount account;
259  walletdb.ReadAccount(strAccount, account);
260 
261  bool bKeyUsed = false;
262 
263  // Check if the current key has been used
264  if (account.vchPubKey.IsValid())
265  {
266  CScript scriptPubKey;
267  scriptPubKey.SetDestination(account.vchPubKey.GetID());
268  for (map<uint256, CWalletTx>::iterator it = pwalletMain->mapWallet.begin();
269  it != pwalletMain->mapWallet.end() && account.vchPubKey.IsValid();
270  ++it)

272  const CWalletTx& wtx = (*it).second;
273  BOOST_FOREACH(const CTxOut& txout, wtx.vout)
274  if (txout.scriptPubKey == scriptPubKey)
275  bKeyUsed = true;
276  }
277  }
278 
279  // Generate a new key
280  if (!account.vchPubKey.IsValid() || bForceNew || bKeyUsed)
281  {
282  if (!pwalletMain->GetKeyFromPool(account.vchPubKey))
283  throw JSONRPCError(RPC_WALLET_KEYPOOL_RAN_OUT, "Error: Keypool ran out, please call keypoolrefill first");
284 
285  pwalletMain->SetAddressBookName(account.vchPubKey.GetID(), strAccount);
286  walletdb.WriteAccount(strAccount, account);
287  }

289  return CIonAddress(account.vchPubKey.GetID());
