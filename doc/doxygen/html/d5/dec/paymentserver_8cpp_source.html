48 // This stores payment requests received before
49 // the main GUI window is up and ready to ask the user
50 // to send payment.

52 static QStringList savedPaymentRequests;

54 //
55 // Sending to the server is done synchronously, at startup.
56 // If the server isn't already running, startup continues,
57 // and the items in savedPaymentRequest will be handled
58 // when uiReady() is called.
59 //
60 bool PaymentServer::ipcSendCommandLine()
61 {
62  bool fResult = false;
63 
64  const QStringList& args = qApp->arguments();
65  for (int i = 1; i < args.size(); i++)
66  {
67  if (!args[i].startsWith(ION_IPC_PREFIX, Qt::CaseInsensitive))
68  continue;
69  savedPaymentRequests.append(args[i]);
70  }
71 
72  foreach (const QString& arg, savedPaymentRequests)
73  {
74  QLocalSocket* socket = new QLocalSocket();
75  socket->connectToServer(ipcServerName(), QIODevice::WriteOnly);
76  if (!socket->waitForConnected(ION_IPC_CONNECT_TIMEOUT))
77  return false;
78 
79  QByteArray block;
80  QDataStream out(&block, QIODevice::WriteOnly);
81  out.setVersion(QDataStream::Qt_4_0);
82  out << arg;
83  out.device()->seek(0);
84  socket->write(block);
85  socket->flush();
86 
87  socket->waitForBytesWritten(ION_IPC_CONNECT_TIMEOUT);
88  socket->disconnectFromServer();
89  delete socket;
90  fResult = true;
91  }
92  return fResult;
93 }

113 bool PaymentServer::eventFilter(QObject *object, QEvent *event)
114 {
115  // clicking on ion: URLs creates FileOpen events on the Mac:
116  if (event->type() == QEvent::FileOpen)
117  {
118  QFileOpenEvent* fileEvent = static_cast<QFileOpenEvent*>(event);
119  if (!fileEvent->url().isEmpty())
120  {
121  if (saveURIs) // Before main window is ready:
122  savedPaymentRequests.append(fileEvent->url().toString());
123  else
124  emit receivedURI(fileEvent->url().toString());
125  return true;
126  }
127  }

139 void PaymentServer::handleURIConnection()
140 {
141  QLocalSocket *clientConnection = uriServer->nextPendingConnection();
142 
143  while (clientConnection->bytesAvailable() < (int)sizeof(quint32))
144  clientConnection->waitForReadyRead();
145 
146  connect(clientConnection, SIGNAL(disconnected()),
147  clientConnection, SLOT(deleteLater()));
148 
149  QDataStream in(clientConnection);
150  in.setVersion(QDataStream::Qt_4_0);
151  if (clientConnection->bytesAvailable() < (int)sizeof(quint16)) {
152  return;
153  }
154  QString message;
155  in >> message;
156 
157  if (saveURIs)
158  savedPaymentRequests.append(message);
159  else
160  emit receivedURI(message);
161 }
