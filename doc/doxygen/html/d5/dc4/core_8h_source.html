162  bool IsDust(int64_t MIN_RELAY_TX_FEE) const
163  {
164  // "Dust" is defined in terms of CTransaction::nMinRelayTxFee,
165  // which has units satoshis-per-kilobyte.
166  // If you'd pay more than 1/3 in fees
167  // to spend something, then we consider it dust.
168  // A typical txout is 34 bytes big, and will
169  // need a CTxIn of at least 148 bytes to spend,
170  // so dust is a txout less than 546 satoshis
171  // with default nMinRelayTxFee.
172  return ((nValue*1000)/(3*((int)GetSerializeSize(SER_DISK,0)+148)) < MIN_RELAY_TX_FEE);
173  }
