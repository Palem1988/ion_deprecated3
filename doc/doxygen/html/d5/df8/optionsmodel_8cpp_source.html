157 QVariant OptionsModel::data(const QModelIndex & index, int role) const
158 {
159  if(role == Qt::EditRole)
160  {
161  QSettings settings;
162  switch(index.row())
163  {
164  case StartAtStartup:
165  return GUIUtil::GetStartOnSystemStartup();
166  case MinimizeToTray:
167  return fMinimizeToTray;
168  case MapPortUPnP:
169 #ifdef USE_UPNP
170  return settings.value("fUseUPnP");
171 #else
172  return false;
173 #endif
174  case MinimizeOnClose:
175  return fMinimizeOnClose;
176 
177  // default proxy
178  case ProxyUse:
179  return settings.value("fUseProxy", false);
180  case ProxyIP: {
181  // contains IP at index 0 and port at index 1
182  QStringList strlIpPort = settings.value("addrProxy").toString().split(":", QString::SkipEmptyParts);
183  return strlIpPort.at(0);
184  }
185  case ProxyPort: {
186  // contains IP at index 0 and port at index 1
187  QStringList strlIpPort = settings.value("addrProxy").toString().split(":", QString::SkipEmptyParts);
188  return strlIpPort.at(1);
189  }
190  case ProxySocksVersion:
191  return settings.value("nSocksVersion", 5);
192 
193 #ifdef ENABLE_WALLET
194  case Fee:
195  // Attention: Init() is called before nTransactionFee is set in AppInit2()!
196  // To ensure we can change the fee on-the-fly update our QSetting when
197  // opening OptionsDialog, which queries Fee via the mapper.
198  if (nTransactionFee != settings.value("nTransactionFee").toLongLong())
199  settings.setValue("nTransactionFee", (qint64)nTransactionFee);
200  // Todo: Consider to revert back to use just nTransactionFee here, if we don't want
201  // -paytxfee to update our QSettings!
202  return settings.value("nTransactionFee");
203  case ReserveBalance:
204  return QVariant((qint64) nReserveBalance);
205 #endif
206  case DisplayUnit:
207  return nDisplayUnit;
208  case Language:
209  return settings.value("language");
210  case CoinControlFeatures:
211  return fCoinControlFeatures;
212  case StashedsendRounds:
213  return QVariant(nStashedsendRounds);
214  case AnonymizeIonAmount:
215  return QVariant(nAnonymizeIonAmount);
216  case UseBlackTheme:
217  return QVariant(fUseBlackTheme);
218  default:
219  return QVariant();
220  }
221  }
222  return QVariant();
223 }

226 bool OptionsModel::setData(const QModelIndex & index, const QVariant & value, int role)
227 {
228  bool successful = true; /* set to false on parse error */
229  if(role == Qt::EditRole)
230  {
231  QSettings settings;
232  switch(index.row())
233  {
234  case StartAtStartup:
235  successful = GUIUtil::SetStartOnSystemStartup(value.toBool());
236  break;
237  case MinimizeToTray:
238  fMinimizeToTray = value.toBool();
239  settings.setValue("fMinimizeToTray", fMinimizeToTray);
240  break;
241  case MapPortUPnP: // core option - can be changed on-the-fly
242  settings.setValue("fUseUPnP", value.toBool());
243  MapPort(value.toBool());
244  break;
245  case MinimizeOnClose:
246  fMinimizeOnClose = value.toBool();
247  settings.setValue("fMinimizeOnClose", fMinimizeOnClose);
248  break;
249 
250  // default proxy
251  case ProxyUse:
252  if (settings.value("fUseProxy") != value) {
253  settings.setValue("fUseProxy", value.toBool());
254  setRestartRequired(true);
255  }
256  break;
257  case ProxyIP: {
258  // contains current IP at index 0 and current port at index 1
259  QStringList strlIpPort = settings.value("addrProxy").toString().split(":", QString::SkipEmptyParts);
260  // if that key doesn't exist or has a changed IP
261  if (!settings.contains("addrProxy") || strlIpPort.at(0) != value.toString()) {
262  // construct new value from new IP and current port
263  QString strNewValue = value.toString() + ":" + strlIpPort.at(1);
264  settings.setValue("addrProxy", strNewValue);
265  setRestartRequired(true);
266  }
267  }
268  break;
269  case ProxyPort: {
270  // contains current IP at index 0 and current port at index 1
271  QStringList strlIpPort = settings.value("addrProxy").toString().split(":", QString::SkipEmptyParts);
272  // if that key doesn't exist or has a changed port
273  if (!settings.contains("addrProxy") || strlIpPort.at(1) != value.toString()) {
274  // construct new value from current IP and new port
275  QString strNewValue = strlIpPort.at(0) + ":" + value.toString();
276  settings.setValue("addrProxy", strNewValue);
277  setRestartRequired(true);
278  }
279  }
280  break;
281  case ProxySocksVersion: {
282  if (settings.value("nSocksVersion") != value) {
283  settings.setValue("nSocksVersion", value.toInt());
284  setRestartRequired(true);
285  }
286  }
287  break;
288 #ifdef ENABLE_WALLET
289  case Fee: // core option - can be changed on-the-fly
290  // Todo: Add is valid check and warn via message, if not
291  nTransactionFee = value.toLongLong();
292  settings.setValue("nTransactionFee", (qint64) nTransactionFee);
293  emit transactionFeeChanged(nTransactionFee);
294  break;
295  case ReserveBalance:
296  nReserveBalance = value.toLongLong();
297  settings.setValue("nReserveBalance", (qint64) nReserveBalance);
298  emit reserveBalanceChanged(nReserveBalance);
299  break;
300 #endif
301  case DisplayUnit:
302  nDisplayUnit = value.toInt();
303  settings.setValue("nDisplayUnit", nDisplayUnit);
304  emit displayUnitChanged(nDisplayUnit);
305  break;
306  case Language:
307  if (settings.value("language") != value) {
308  settings.setValue("language", value);
309  setRestartRequired(true);
310  }
311  break;
312  case CoinControlFeatures:
313  fCoinControlFeatures = value.toBool();
314  settings.setValue("fCoinControlFeatures", fCoinControlFeatures);
315  emit coinControlFeaturesChanged(fCoinControlFeatures);
316  break;
317  case UseBlackTheme:
318  fUseBlackTheme = value.toBool();
319  settings.setValue("fUseBlackTheme", fUseBlackTheme);
320  break;
321  case StashedsendRounds:
322  nStashedsendRounds = value.toInt();
323  settings.setValue("nStashedsendRounds", nStashedsendRounds);
324  emit stashedsendRoundsChanged(nStashedsendRounds);
325  break;
326  case AnonymizeIonAmount:
327  nAnonymizeIonAmount = value.toInt();
328  settings.setValue("nAnonymizeIonAmount", nAnonymizeIonAmount);
329  emit AnonymizeIonAmountChanged(nAnonymizeIonAmount);
330  break;
331  default:
332  break;
333  }
334  }
335  emit dataChanged(index, index);
336 
337  return successful;
338 }

340 bool OptionsModel::getProxySettings(QNetworkProxy& proxy) const
341 {
342  // Directly query current base proxy, because
343  // GUI settings can be overridden with -proxy.
344  proxyType curProxy;
345  if (GetProxy(NET_IPV4, curProxy)) {
346  if (curProxy.second == 5) {
347  proxy.setType(QNetworkProxy::Socks5Proxy);
348  proxy.setHostName(QString::fromStdString(curProxy.first.ToStringIP()));
349  proxy.setPort(curProxy.first.GetPort());
350 
351  return true;
352  }
353  else
354  return false;
355  }
356  else
357  proxy.setType(QNetworkProxy::NoProxy);
358 
359  return true;
360 }
