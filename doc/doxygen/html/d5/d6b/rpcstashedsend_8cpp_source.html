24 void SendMoney(const CTxDestination &address, CAmount nValue, CWalletTx& wtxNew, AvailableCoinsType coin_type=ALL_COINS)
25 {
26  // Check amount
27  if (nValue <= 0)
28  throw JSONRPCError(RPC_INVALID_PARAMETER, "Invalid amount");
29 
30  if (nValue > pwalletMain->GetBalance())
31  throw JSONRPCError(RPC_WALLET_INSUFFICIENT_FUNDS, "Insufficient funds");
32 
33  string strError;
34  if (pwalletMain->IsLocked())
35  {
36  strError = "Error: Wallet locked, unable to create transaction!";
37  LogPrintf("SendMoney() : %s", strError);
38  throw JSONRPCError(RPC_WALLET_ERROR, strError);
39  }
40 
41  // Parse ion address
42  CScript scriptPubKey = GetScriptForDestination(address);
43 
44  // Create and send the transaction
45  CReserveKey reservekey(pwalletMain);
46  CAmount nFeeRequired;
47  std::string sNarr;
48  if (!pwalletMain->CreateTransaction(scriptPubKey, nValue, sNarr, wtxNew, reservekey, nFeeRequired, NULL))
49  {
50  if (nValue + nFeeRequired > pwalletMain->GetBalance())
51  strError = strprintf("Error: This transaction requires a transaction fee of at least %s because of its amount, complexity, or use of recently received funds!", FormatMoney(nFeeRequired));
52  LogPrintf("SendMoney() : %s\n", strError);
53  throw JSONRPCError(RPC_WALLET_ERROR, strError);
54  }
55  if (!pwalletMain->CommitTransaction(wtxNew, reservekey))
56  throw JSONRPCError(RPC_WALLET_ERROR, "Error: The transaction was rejected! This might happen if some of the coins in your wallet were already spent, such as if you used a copy of wallet.dat and coins were spent in the copy but not marked as spent here.");
57 }
