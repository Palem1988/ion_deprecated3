106 bool ReadHTTPRequestLine(std::basic_istream<char>& stream, int &proto,
107  string& http_method, string& http_uri)
108 {
109  string str;
110  getline(stream, str);
111 
112  // HTTP request line is space-delimited
113  vector<string> vWords;
114  boost::split(vWords, str, boost::is_any_of(" "));
115  if (vWords.size() < 2)
116  return false;
117 
118  // HTTP methods permitted: GET, POST
119  http_method = vWords[0];
120  if (http_method != "GET" && http_method != "POST")
121  return false;
122 
123  // HTTP URI must be an absolute path, relative to current host
124  http_uri = vWords[1];
125  if (http_uri.size() == 0 || http_uri[0] != '/')
126  return false;
127 
128  // parse proto, if present
129  string strProto = "";
130  if (vWords.size() > 2)
131  strProto = vWords[2];
132 
133  proto = 0;
134  const char *ver = strstr(strProto.c_str(), "HTTP/1.");
135  if (ver != NULL)
136  proto = atoi(ver+7);
137 
138  return true;
139 }

156 int ReadHTTPHeaders(std::basic_istream<char>& stream, map<string, string>& mapHeadersRet)
157 {
158  int nLen = 0;
159  while (true)
160  {
161  string str;
162  std::getline(stream, str);
163  if (str.empty() || str == "\r")
164  break;
165  string::size_type nColon = str.find(":");
166  if (nColon != string::npos)
167  {
168  string strHeader = str.substr(0, nColon);
169  boost::trim(strHeader);
170  boost::to_lower(strHeader);
171  string strValue = str.substr(nColon+1);
172  boost::trim(strValue);
173  mapHeadersRet[strHeader] = strValue;
174  if (strHeader == "content-length")
175  nLen = atoi(strValue.c_str());
176  }
177  }
178  return nLen;
179 }

182 int ReadHTTPMessage(std::basic_istream<char>& stream, map<string,
183  string>& mapHeadersRet, string& strMessageRet,
184  int nProto, size_t max_size)
185 {
186  mapHeadersRet.clear();
187  strMessageRet = "";
188 
189  // Read header
190  int nLen = ReadHTTPHeaders(stream, mapHeadersRet);
191  if (nLen < 0 || (size_t)nLen > max_size)
192  return HTTP_INTERNAL_SERVER_ERROR;
193 
194  // Read message
195  if (nLen > 0)
196  {
197  vector<char> vch;
198  size_t ptr = 0;
199  while (ptr < (size_t)nLen)
200  {
201  size_t bytes_to_read = std::min((size_t)nLen - ptr, POST_READ_SIZE);
202  vch.resize(ptr + bytes_to_read);
203  stream.read(&vch[ptr], bytes_to_read);
204  if (!stream) // Connection lost while reading
205  return HTTP_INTERNAL_SERVER_ERROR;
206  ptr += bytes_to_read;
207  }
208  strMessageRet = string(vch.begin(), vch.end());
209  }
210 
211  string sConHdr = mapHeadersRet["connection"];
212 
213  if ((sConHdr != "close") && (sConHdr != "keep-alive"))
214  {
215  if (nProto >= 1)
216  mapHeadersRet["connection"] = "keep-alive";
217  else
218  mapHeadersRet["connection"] = "close";
219  }
220 
221  return HTTP_OK;
222 }
